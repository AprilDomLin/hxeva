[文档]def calc_dis(x1: float or int or np.ndarray, y1: float or int or np.ndarray,
             x2: float or int or np.ndarray, y2: float or int or np.ndarray) -> np.ndarray or float:
    """欧式距离
    
    :math:`Distance = \\sqrt{{x_1 - x_2}^2 + {y_1 - y_2}^2}`
    
    Args:
        x1 : 起点的 x轴/横向 坐标值
        y1 : 起点的 y轴/纵向 坐标值
        x2 : 终点的 x轴/横向 坐标值
        y2 : 终点的 y轴/纵向 坐标值
    
    Returns:
        欧式距离
    """
    return np.sqrt((x1 - x2) ** 2 + (y1 - y2) ** 2)




[文档]def calc_rhu(tmp: np.ndarray or float, dpt: np.ndarray or float) -> np.ndarray or float:
    """ 计算空气相对湿度
    
    基于温度(K)和露点温度(K)计算空气的相对湿度。无量纲，理论值域范围为0 ~ 1。
    
    :math:`rhu = \\exp{(17.67 * (dpt - 273.15) / (dpt - 29.65) - 17.67 * (tmp - 273.15) / (tmp - 29.65))}`
    
    Args:
        tmp : Temperature, unit: K
        dpt : DewPoint Temperature, unit: K

    Returns:
        空气相对湿度
    """
    return np.exp(17.67 * (dpt - 273.15) / (dpt - 29.65)) / np.exp(17.67 * (tmp - 273.15) / (tmp - 29.65))




[文档]def dewind(ws: np.ndarray or float, wd: np.ndarray or float) -> tuple:
    """ 风场(正交)分解
    
    将风速和风向分解为 u / v 风。

    Args:
        ws : 风速值, 单位为 m/s
        wd : 风向值, 单位为 °

    Returns:
        u风(矢量) 和 v风(矢量)
    """
    u = ws * np.cos(wd * np.pi / 360)
    v = ws * np.sin(wd * np.pi / 360)
    return u, v




[文档]def comwind(u: np.ndarray or float, v: np.ndarray or float) -> tuple:
    """ 合成风场
    
    由 u / v  风，合成风速和风向

    Args:
        u : u风, 单位为 m/s
        v : v风, 单位为 m/s

    Returns:
        风速(单位: m/s) 和 风向(单位: °)
    """

    if u is None or v is None:
        return None, None
    ws = np.sqrt(np.square(u) + np.square(v))
    wd = 270.0 - np.arctan2(v, u) * 180.0 / np.pi
    wd = wd % 360.0
    return ws, wd
