ContingencyTable = namedtuple("ContingencyTable", ["tp", "fp", "fn", "tn"], defaults=(None, None, None, None))
"""联列表。基于联列表分类的结果，包含 :obj:`tp` 、:obj:`fp` 、:obj:`fn` 和 :obj:`tn` 。

   ===============  =============  =============
    观测  \\  预测        正例             反例
   ===============  =============  =============
       正例              tp              fn
       反例              fp              tn
   ===============  =============  =============

"""
ContingencyTable.tp.__doc__ = "int: 默认为 ``None`` 。 命中个数，若为-1，则不存在对应等级分类。"
ContingencyTable.fp.__doc__ = "int: 默认为 ``None`` 。 空报/误报个数，若为-1，则不存在对应等级分类。"
ContingencyTable.fn.__doc__ = "int: 默认为 ``None`` 。 漏报个数，若为-1，则不存在对应等级分类。"
ContingencyTable.tn.__doc__ = "int: 默认为 ``None`` 。 否定正确个数，若为-1，则不存在对应等级分类。"


def _score_check(func):
    @wraps(func)
    def _default(tables: ContingencyTable):
        if tables.count(None) > 0:
            Warning(f"评分计算需要先对truth 和 test数据进行分类操作，ContingencyTable数据为空!")
            return np.nan
        if tables.count(-1) == 4:
            Warning(f"ContingencyTable数据为-1，不存在该等级的数据划分，请选择其他等级进行评分计算！")
            return np.nan
        return func(tables)
    return _default


@_score_check
def calc_acc(tables: ContingencyTable) -> float:
    """ 准确率 / 晴雨评分
    
    :math:`Accuracy = \\frac{tp + tn}{tp + fp + fn + tn}`
    
    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

    Returns:
        准确率 / 晴雨评分。当预报无降水时或未对应等级降水时，返回 :obj:`NaN` 。
    """
    if sum(tables):
        return (tables.tp + tables.tn) / sum(tables)
    Warning(f"无对应等级分类情况，请检查前序操作！")
    return np.nan


@_score_check
def calc_bias(tables: ContingencyTable) -> float:
    """ 偏差值
    
    :math:`Bias = \\left|\\frac{tp + tn}{tp + fn} - 1\\right|`

    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

    Returns:
        偏差值。当预报无降水时或未对应等级降水时，返回 :obj:`NaN` 。
    """
    if tables.tp + tables.fn:
        return abs((tables.tp + tables.fp) / (tables.tp + tables.fn) - 1)
    Warning(f"无对应等级分类情况，请检查前序操作！")
    return np.nan


@_score_check
def calc_ets(tables: ContingencyTable) -> float:
    """ ETS技巧评分
    
    :math:`ETS = \\frac{tp -ra}{tp + fp + fn + tn - ra}`

    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

    Returns:
        ETS技巧评分。当预报无降水时或未对应等级降水时，返回 :obj:`NaN` 。
    """
    ra = calc_ra(tables)
    if not np.isnan(ra):
        if tables.tp + tables.fp + tables.fn - ra:
            return tables.tp / (tables.tp + tables.fp + tables.fn - ra)
    Warning(f"无对应等级分类情况，请检查前序操作！")
    return np.nan


@_score_check
def calc_far(tables: ContingencyTable) -> float:
    """ 空报率 / 误报率

    :math:`FAR = \\frac{fp}{tp + fp}`

    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

    Returns:
        空报率 / 误报率。当预报无降水时或未对应等级降水时，返回 :obj:`NaN` 。
    """
    if tables.fp + tables.tp:
        return tables.fp / (tables.fp + tables.tp)
    Warning(f"无对应等级分类情况，请检查前序操作！")
    return np.nan


@_score_check
def calc_hss(tables: ContingencyTable) -> float:
    """ HSS技巧评分
    
    :math:`expect = \\frac{(tp + fn) * (tp + fp) + (tn + fn) * (tn + fp)}{tp + fp + fn + tn}`
    
    :math:`HSS = \\frac{tp + tn - expect}{tp + fp + fn + tn - expect}`
    
    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

    Returns:
        HSS技巧评分。当预报无降水时或未对应等级降水时，返回 :obj:`NaN` 。
    """
    ra = calc_ra(tables)
    rd = calc_rd(tables)
    if not np.isnan(ra) and not np.isnan(rd):
        if sum(tables) - ra - rd:
            return (tables.tp + tables.tn - ra - rd) / (sum(tables) - ra - rd)
    Warning(f"无对应等级分类情况，请检查前序操作！")
    return np.nan

    return np.nan

@_score_check
def calc_ra(tables: ContingencyTable) -> float:
    """ ra

    :math:`ra = \\frac{(tp + fp) * (tp + fn)}{tp + fp + fn + tn}`

    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

    Returns:
        ra。当预报无降水时或未对应等级降水时，返回 :obj:`NaN` 。
    """
    if sum(tables):
        return (tables.tp + tables.fp) * (tables.tp + tables.fn) / sum(tables)
    Warning(f"无对应等级分类情况，请检查前序操作！")
    return np.nan


@_score_check
def calc_rd(tables: ContingencyTable) -> float:
    """ rd
    
    :math:`ra = \\frac{(tn + fp) * (tn + fn)}{tp + fp + fn + tn}`

    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

    Returns:
        rd。当预报无降水时或未对应等级降水时，返回 :obj:`NaN` 。
    """
    if sum(tables):
        return (tables.tn + tables.fp) * (tables.tn + tables.fn) / sum(tables)
    Warning(f"无对应等级分类情况，请检查前序操作！")
    return np.nan

    return np.nan


@_score_check
def calc_ts(tables: ContingencyTable) -> float:
    """TS评分
    
    :math:`SR = \\frac{tp}{tp + fp + fn}`

    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

[文档]    @classmethod
    def calc(cls, truth: np.ndarray, test: np.ndarray, **kwargs: Any) -> None:
        """算法人员调用进行相关评分计算

        Args:
            truth: 真实值数组
            test: 待测试值数组

        Keyword Args:
            threshold(list): 默认为 ``[0, 0.1, 2.5, 8.0, 16.0, 150]``。 分类阈值表，阈值列表最小值和最大值以外的值，认为是异常值，不纳入计算。
            level(list[int]): 默认为 ``[0, 1, ... len(threshold) - 1]``。 需要计算评分的等级。
            names(list[str]): 默认为 ``["TS", "PO", "FAR", "POD", "ACC", "HSS"]``。需要计算的评分内容。
            method(:obj:`str`, optional): 默认为 ``binary``。参数可选择 ``binary`` (二分类), ``multi`` (多分类) 和 ``grade`` (多级)。
            window_size(tuple): 默认为 ``None``。领域匹配半径格点数目。
            acc(int): 默认为 ``4``。计算结果的精度值，即保留有效小数点位数
            save_path(str): 默认为 ``./scores.csv``。评分结果csv文件的保存路径。
        """
        threshold = kwargs.pop("level", [0, 0.1, 2.5, 8.0, 16.0, 150])
        level = kwargs.pop("level", [i for i in range(len(threshold))])
        names = kwargs.pop("names", ["TS", "PO", "FAR", "POD", "ACC", "HSS"])
        method = kwargs.pop("method", "binary")
        window_size = kwargs.pop("window_size", None)
        acc = kwargs.pop("acc", 4)
        save_path = kwargs.pop("save_path", "./scores.csv")

        res = cls.scores(truth=truth, test=test, threshold=threshold, level=level, names=names, method=method,
                         window_size=window_size, acc=acc)
        
        with open(save_path, mode="w+") as f:
            f.write(",".join(["level"] + names) + "\n")
            print("level\t" + "\t".join(names))
            for lvl in level:
                tmp = list()
                for name in names:
                    tmp_data = res[f"{lvl}"][name]
                    tmp.append(f"{tmp_data}")
                f.write(",".join([f"{level[lvl]}"] + tmp) + "\n")
                print(f"≥ {level[lvl]}\t" + "\t".join(tmp))
        print(f">>>Scores Calculated Done! Result has been saved in \"{save_path}\"!")

    def classify(cls, truth: np.ndarray, test: np.ndarray, level: Union[int, float],
                 exist_flag: Optional[np.ndarray] = None, method: str = "binary",
                 window_size: Optional[Tuple[int, int]] = None) -> ContingencyTable:
        """基于联列表讲数据分为 :obj:`tp`、:obj:`fp`、:obj:`fn` 和 :obj:`tn`.

        .. note::
            TP-命中   FP-误报/空报   FN-漏报   D-否定正确

            二分类 binary method
                ==================  ===================    ===================
                  观测  \\  预测       FCT >= threshold       FCT <  threshold
                ==================  ===================    ===================
                 OBS >= threshold         TP                      FN
                 OBS <  threshold         FP                      TN
                ==================  ===================    ===================

            多分类 multi method
                ==================  ===================    ===================
                  观测  \\  预测        FCT == threshold       FCT =! threshold
                ==================  ===================    ===================
                 OBS == threshold         TP                      FN
                 OBS =! threshold         FP                      TN
                ==================  ===================    ===================

            多级别 grade method
                ==================  ===================    ===================
                  观测  \\  预测        FCT == threshold       FCT <  threshold
                ==================  ===================    ===================
                 OBS == threshold         TP                      FN
                 OBS <  threshold         FP                      TN
                ==================  ===================    ===================

        Args:
            truth: 真实值数组 或 转变数组
            test: 待测试值数组 或 转变数组
            level: 分类等级
            exist_flag: 默认为 ``None`` 。 数据存在与否的标记，shape同truth和test。数组值为True时, 表明truth和test均存在，否则存在缺失。
            method: 默认为 ``binary`` 。参数可选择 ``binary`` (二分类), ``multi`` (多分类) 和 ``grade`` (多级)。
            window_size: 默认为 ``None`` 。 领域匹配窗口直径格点数，若为None，则不进行领域匹配。 【注：当前仅支持 binary 二分类方法时使用】

        Returns:
            联列表分类结果。TP, FP, FN, TN

        """
        # 默认降水 0 ~ 999 mm 区间内, 同时排除 NaN 数据
        if exist_flag is None:
            exist_flag = np.where((0 <= truth) & (truth < 999) & (0 <= test) & (test < 999), True, False)
        if np.sum(exist_flag):
            if method == "binary":
                if isinstance(window_size, tuple):
                    # 更正 test 正例，truth匹配领域范围最大值，达到提高tp、降低fp的效果。
                    tmp_truth = cls.expand(truth, exist_flag=exist_flag, window_size=window_size, method="max")
                    truth[test >= level] = tmp_truth[test >= level]
                a = (truth >= level) & exist_flag
                b = (truth < level) & exist_flag
                c = (test >= level) & exist_flag
                d = (test < level) & exist_flag
            elif method == "multi":
                a = (truth == level) & exist_flag
                b = (truth != level) & exist_flag
                c = (test == level) & exist_flag
                d = (test != level) & exist_flag
            elif method == "grade":
                a = (truth == level) & exist_flag
                b = (truth < level) & exist_flag
                c = (test == level) & exist_flag
                d = (test < level) & exist_flag
            else:
                raise KeyError("The classification method only supports 'binary', 'multi' and 'grade'!")
            return ContingencyTable(tp=np.sum(a & c).astype(int), fp=np.sum(b & c).astype(int),
                                    fn=np.sum(a & d).astype(int), tn=np.sum(b & d).astype(int))
        else:
            return ContingencyTable(tp=-1, fp=-1, fn=-1, tn=-1)

    def expand(cls, data: np.ndarray, exist_flag: np.ndarray = None,
               window_size: Tuple[int, int] = (11, 11), method: str = "max") -> np.ndarray:
        """模糊空间领域匹配。

        Args:
            data : 待模糊空间领域匹配的数组
            exist_flag : 默认为 ``None``。 数据存在与否的标记，shape同truth和test。数组值为True时, 表明truth和test均存在，否则存在缺失。
            window_size : 默认为 ``11``。领域匹配的窗口直径格点数目，非距离长度或经纬度度数。
            method: 默认为 ``max``。领域匹配的方法， 仅支持 ``max`` 和 ``mean`` 两种。

        Returns:
            模糊空间领域匹配的结果。其 shape 与 `data` 相同。
        """
        if exist_flag is None:
            exist_flag = np.where((0 <= data) & (data < 999), True, False)
        data[~exist_flag] = np.nan
        new_data = pool2d(data=data, window_size=window_size, stride=(1, 1), method=method, pad=True)
        return new_data

[文档]    @classmethod
    def scores(cls, truth: np.ndarray, test: np.ndarray, **kwargs: Any) -> dict:
        """更灵活的评分调用方式，类似calc。但没有封装结果保存与打印。

        Args:
            truth: 真实值数组

            test: 待测试值数组

        Keyword Args:
            threshold(list): 默认为 ``[0, 0.1, 2.5, 8.0, 16.0, 150]``。 分类阈值表，阈值列表最小值和最大值以外的值，认为是异常值，不纳入计算。
            level(list[int]): 默认为 ``[0, 1, ... len(threshold) - 1]``。 需要计算评分的等级。
            names(list[str]): 默认为 ``["TS", "PO", "FAR", "POD", "ACC", "HSS"]``。需要计算的评分内容。
            method(:obj:`str`, optional): 默认为 ``binary``。参数可选择 ``binary`` (二分类), ``multi`` (多分类) 和 ``grade`` (多级)。
            window_size(tuple): 默认为 ``None``。领域匹配半径格点数目。
            acc(int): 默认为 ``4``。计算结果的精度值，即保留有效小数点位数

        Returns:
            各等级的不同评分结果。
        """
        # Keyword Args 默认设置
        threshold = kwargs.pop("level", [0, 0.1, 2.5, 8.0, 16.0, 150])
        level = kwargs.pop("level", [i for i in range(len(threshold))])
        names = kwargs.pop("names", ["TS", "PO", "FAR", "POD", "ACC", "HSS"])
        method = kwargs.pop("method", "binary")
        window_size = kwargs.pop("window_size", None)
        acc = kwargs.pop("acc", 4)
        
        # 根据阈值分类原始数据
        truth = cls.transform(data=truth, threshold=threshold)
        test = cls.transform(data=test, threshold=threshold)
        exist_flag = kwargs.get("exist_flag", np.where((truth < len(threshold)) & (test < len(threshold)), True, False))
        
        # 多指标、多等级评分参数
        if isinstance(level, int):
            level = [level]
        
        res = dict()
        for lvl in level:
            tables = cls.classify(truth=truth, test=test, exist_flag=exist_flag, level=lvl,
                                  method=method, window_size=window_size)
            tmp = dict()
            for name in names:
                tmp[name] = np.round(getattr(cls, name.lower())(tables), acc)
            res[f"{lvl}"] = tmp.copy()
        return res

        """ 成功率
