def _space_check(func):
    @wraps(func)
    def _default(truth, test, **kwargs):
        kwargs.setdefault("exist_flag", np.ones_like(kwargs["truth"])).astype(int)
        kwargs.setdefault("threshold", [0.1, 999])
        kwargs.setdefault("min_points", 15)
        if not isinstance(truth, Property):
            truth = Space.props(truth, **kwargs)
        if not isinstance(test, Property):
            test = Space.props(test, **kwargs)
        return func(truth, test, **kwargs)
    
    return _default


def calc_S(truth: Property, test: Property) -> float:
    """结构分值计算

    Args:
        truth : 真值数据区域划分数据
        test :  测试值数据区域划分数据

    Returns:
        结构得分(S)

    References:
        [1] Wernli, H., Paulat, M., Hagen, M., & Frei, C.M. (2008). SAL—A Novel Quality Measure for the Verification of Quantitative Precipitation Forecasts. Monthly Weather Review, 136, 4470-4487.
    """
    assert isinstance(truth, Property), TypeError(f"Expected \"Property\", but got the type of truth: {type(truth)}")
    assert isinstance(test, Property), TypeError(f"Expected \"Property\", but got the type of test: {type(test)}")
    
    def scaled_volume(tmp: Property) -> float:
        """Calculate Scaled Volume

        Args:
            tmp
        """
        v = 0
        for prop in tmp.props:
            rn = prop["mean_intensity"] * prop["area"]
            v += rn ** 2 / prop["max_intensity"]
        return v / (tmp.all_prop["mean_intensity"] * tmp.all_prop["area"])
    
    v_truth = scaled_volume(truth)
    v_test = scaled_volume(test)
    return (v_test - v_truth) / (v_test + v_truth) / 0.5


def calc_A(truth: Property, test: Property) -> float:
    """振幅分值计算

    Args:
        truth : 真值数据区域划分数据
        test :  测试值数据区域划分数据

    Returns:
        振幅得分(A)

    References:
        [1] Wernli, H., Paulat, M., Hagen, M., & Frei, C.M. (2008). SAL—A Novel Quality Measure for the Verification of Quantitative Precipitation Forecasts. Monthly Weather Review, 136, 4470-4487.
    """
    d_truth = truth.all_prop["mean_intensity"]
    d_test = test.all_prop["mean_intensity"]
    return (d_test - d_truth) / ((d_test + d_truth) * 0.5)


def calc_L(truth: Property, test: Property) -> float:
    """ 落区分值计算

    Args:
        truth : 真值数据区域划分数据
        test :  测试值数据区域划分数据

    Returns:
        落区得分(L)

    References:
        [1] Wernli, H., Paulat, M., Hagen, M., & Frei, C.M. (2008). SAL—A Novel Quality Measure for the Verification of Quantitative Precipitation Forecasts. Monthly Weather Review, 136, 4470-4487.
    """
    # L1
    # 计算区域内非缺测点的最大距离，这里距离都是用网格点位置而不是经纬度
    convect_union = truth.label + test.label
    convect_union[convect_union < 0] = 1
    tmp_prop = regionprops(convect_union, truth.data)
    part_L1 = calc_dis(*truth.all_prop["centroid"], *test.all_prop["centroid"])
    
    # L2
    def weighted_distance(tmp: Property) -> float:
        """ Calculate Weighted Averaged Distance.
        Calculate the weighted averaged distance between the centers of mass of the individual objects,
        and the center of mass of the total precipitation field

        Args:
            tmp

        Returns:
            Weighted Averaged Distance
        """
        row, col = tmp.all_prop["centroid"]
        distance = reduce(lambda x, y: x + calc_dis(row, col, *y["centroid"]) * y["mean_intensity"] * y["area"],
                          [0] + tmp.props)
        return distance / (tmp.all_prop["mean_intensity"] * tmp.all_prop["area"])
    
    part_L2 = abs(weighted_distance(truth) - weighted_distance(test))
    return (part_L1 + part_L2) / calc_dis(*tmp_prop[0]["bbox"])

[文档]    @classmethod
    def props(cls, data: np.ndarray, **kwargs: Any) -> Property:
        """ 计算图像区域所有落区的属性。

        Args:
            data(:obj:`ndarray`): 需要划分落区的数据矩阵。

        Keyword Args:
            exist_flag(:obj:`ndarray`): 默认为 ``None``。数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
            threshold(list): 默认为 ``[0.1, 999]``。SAL空间检验的边界阈值范围。
            min_points(list): 默认为 ``15``。SAL空间检验的最小落区格点数目。

        Returns:
            落区的属性标签
        """
        exist_flag = kwargs.pop("exist_flag", np.ones_like(data)).astype(int)
        threshold = kwargs.pop("threshold", [0.1, 999])
        min_points = kwargs.pop("min_points", 15)
        
        # 落区划分
        convert = (np.where((threshold[0] <= data) & (data < threshold[1]), 1, 0) * exist_flag)
        convert = label(convert, connectivity=2, background=0)
        tmp_props = regionprops(convert, data)
        
        # 去除小于 min_points点数的 落区
        if tmp_props:
            props = []
            for prop in tmp_props:
                if prop["area"] >= min_points:
                    props.append(prop)
                else:
                    flag = np.where(convert == prop["label"], 0, 1)
                    convert *= flag
            convert[convert > 0] = 1
            return Property(data=data, label=convert, all_prop=regionprops(convert, data)[0], props=props)
        return Property(data=data, label=convert, all_prop=[], props=[])

        return truth, test

    def SAL(cls, truth: np.ndarray, test: np.ndarray, **kwargs: Any) -> dict:
        """ SAL空间检验

        Args:
            truth : 真实值数组
            test : 待测试值数组

        Keyword Args:
            exist_flag(:obj:`ndarray`): 默认为 ``None``。数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
            threshold(list): 默认为 ``[0.1， 999]``。SAL空间检验的边界阈值范围。
            min_points(list): 默认为 ``15``。SAL空间检验的最小落区格点数目。
            acc(int): 默认为 ``4``。计算结果的精度值，即保留有效小数点位数

        Returns:
            SAL检验结果

        """
        # 原始数据预处理
        exist_flag = kwargs.pop("exist_flag", np.ones_like(truth)).astype(int)
        threshold = kwargs.pop("threshold", [0.1, 999])
        min_points = kwargs.pop("min_points", 15)
        acc = kwargs.pop("acc", 4)
        
        truth = cls.props(truth, exist_flag=exist_flag, threshold=threshold, min_points=min_points)
        test = cls.props(test, exist_flag=exist_flag, threshold=threshold, min_points=min_points)
        
        # SAL
        res = dict(S=round(cls.S(truth, test, exist_flag=exist_flag), acc),
                   A=round(cls.A(truth, test, exist_flag=exist_flag), 4),
                   L=round(cls.L(truth, test, exist_flag=exist_flag), 4))
        return res

    
    @classmethod
    def _windowFSS(cls, data: np.ndarray,
                   i: int, j: int, w_hh: int, w_hw: int,
                   exist_flag: np.ndarray) -> float:
        tmp = data[i - w_hh: i + w_hh + 1, j - w_hw: j + w_hw + 1]
        tmp_flag = exist_flag[i - w_hh: i + w_hh + 1, j - w_hw: j + w_hw + 1]
        if np.sum(tmp_flag):
            return float(np.mean(tmp[tmp_flag]))
        else:
            return np.nan

    def FSS(cls, truth: np.ndarray, test: np.ndarray, **kwargs: Any) -> np.ndarray:
        """FSS(Fraction Skill Score)空间检验。

        Args:
            truth : 真实值数组
            test : 待测试值数组

        Keyword Args:
            window_size(list): 默认为 ``[3, 3]``。卷积窗口宽度的列表，以格点数为单位
            threshold(list): 默认为 ``[0.1]``。事件发生的阈值
            exist_flag(:obj:`ndarray`): 默认为 ``None``。数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。

        Returns:
            shape与threshold相同，根据不同阈值划分的FSS苹果结果
        """
        assert isinstance(truth, np.ndarray), TypeError(f'Expected "np.ndarray",'
                                                        f" but got the type of truth: {type(truth)}")
        assert isinstance(test, np.ndarray), TypeError(f'Expected "np.ndarray",'
                                                       f" but got the type of test: {type(test)}")
        
        assert truth.ndim == 2, ValueError(f"Expected 2-Dim, but got the dimension of the truth: {truth.ndim}")
        assert test.ndim == 2, ValueError(f"Expected 2-Dim, but got the dimension of the test: {test.ndim}")
        
        assert truth.shape == test.shape, ValueError(f"Expected the same shape, but the shape of truth - {truth.shape}"
                                                     f" is different with test - {test.shape} ")
        
        window_size = kwargs.pop("window_size", [3, 3])
        threshold = kwargs.pop("threshold", [0.1])
        exist_flag = kwargs.pop("exist_flag", np.ones_like(truth).astype(np.bool_))
        
        h, w = truth.shape
        wHH, wHW = int((window_size[0] - 1) / 2), int((window_size[1] - 1) / 2)
        fss = np.ones_like(threshold) * np.nan
        for index, val in enumerate(threshold):
            truthHap = np.zeros_like(truth)
            truthHap[truth >= val] = 1
            
            testHap = np.zeros_like(test)
            testHap[test >= val] = 1
            
            a1, a2 = 0, 0
            for i in range(wHH, h - wHH):
                for j in range(wHW, w - wHW):
                    truth_p = cls._windowFSS(truthHap, i, j, wHH, wHW, exist_flag)
                    test_p = cls._windowFSS(testHap, i, j, wHH, wHW, exist_flag)
                    if (not np.isnan(truth_p)) and (not np.isnan(test_p)):
                        a1 += (truth_p - test_p) ** 2
                        a2 += truth_p ** 2 + test_p ** 2
            fss[index] = 1 - a1 / (a2 + 1e-10)
        # 去除结果值
        fss[fss != fss] = np.nan
        return fss
