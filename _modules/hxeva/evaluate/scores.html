ContingencyTable = namedtuple("ContingencyTable", ["tp", "fp", "fn", "tn"], defaults=(None, None, None, None))
"""联列表。基于联列表分类的结果，包含 :obj:`tp` 、:obj:`fp` 、:obj:`fn` 和 :obj:`tn` 。

   ===============  =============  =============
    观测  \\  预测        正例             反例
   ===============  =============  =============
       正例              tp              fn
       反例              fp              tn
   ===============  =============  =============

"""
ContingencyTable.tp.__doc__ = "int: 默认为 ``None`` 。 命中个数，若为-1，则不存在对应等级分类。"
ContingencyTable.fp.__doc__ = "int: 默认为 ``None`` 。 空报/误报个数，若为-1，则不存在对应等级分类。"
ContingencyTable.fn.__doc__ = "int: 默认为 ``None`` 。 漏报个数，若为-1，则不存在对应等级分类。"
ContingencyTable.tn.__doc__ = "int: 默认为 ``None`` 。 否定正确个数，若为-1，则不存在对应等级分类。"


def _score_check(func):
    @wraps(func)
    def _default(tables: ContingencyTable):
        if tables.count(None) > 0:
            Warning(f'As if "Data of Scorer(truth, test)" hasn\'t been classified yet!')
            return Default.MissValue
        if tables.count(-1) > 0:
            return Default.MissValue
        return func(tables)
    
    return _default

def calc_acc(tables: ContingencyTable) -> float:
    """ 准确率 / 晴雨评分
    
    :math:`Accuracy = \\frac{tp + tn}{tp + fp + fn + tn}`
    
    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

    Returns:
        准确率 / 晴雨评分。当预报无降水时，:obj:`Default.NoneValue` 。当tables存在有-1 或 None值，则返回 :obj:`Default.MissValue`。
    """
    if sum(tables):
        return (tables.tp + tables.tn) / sum(tables)
    return Default.NoneValue

def calc_bias(tables: ContingencyTable) -> float:
    """ 偏差值
    
    :math:`Bias = \\left|\\frac{tp + tn}{tp + fn} - 1\\right|`

    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

    Returns:
        偏差值。当预报无降水时，:obj:`Default.NoneValue` 。当tables存在有-1 或 None值，则返回 :obj:`Default.MissValue`。
    """
    if tables.tp + tables.fn:
        return abs((tables.tp + tables.fp) / (tables.tp + tables.fn) - 1)
    return Default.NoneValue

def calc_ets(tables: ContingencyTable) -> float:
    """ ETS技巧评分
    
    :math:`ETS = \\frac{tp -ra}{tp + fp + fn + tn - ra}`

    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

    Returns:
        ETS技巧评分。当预报无降水时，:obj:`Default.NoneValue` 。当tables存在有-1 或 None值，则返回 :obj:`Default.MissValue`。
    """
    ra = calc_ra(tables)
    if ra != Default.MissValue:
        if tables.tp + tables.fp + tables.fn - ra:
            return tables.tp / (tables.tp + tables.fp + tables.fn - ra)
        return Default.NoneValue
    return Default.MissValue

def calc_far(tables: ContingencyTable) -> float:
    """ 空报率 / 误报率

    :math:`FAR = \\frac{fp}{tp + fp}`

    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

    Returns:
        空报率 / 误报率。当预报无降水时，:obj:`Default.NoneValue` 。当tables存在有-1 或 None值，则返回 :obj:`Default.MissValue`。
    """
    if tables.fp + tables.tp:
        return tables.fp / (tables.fp + tables.tp)
    return Default.NoneValue

def calc_hss(tables: ContingencyTable) -> float:
    """ HSS技巧评分
    
    :math:`expect = \\frac{(tp + fn) * (tp + fp) + (tn + fn) * (tn + fp)}{tp + fp + fn + tn}`
    
    :math:`HSS = \\frac{tp + tn - expect}{tp + fp + fn + tn - expect}`
    
    Args:
        tables: 联列表分类结果，包含 :obj:`tp` 、 :obj:`fp` 、 :obj:`fn` 和 :obj:`tn` 的数值。

    Returns:
        HSS技巧评分。当预报无降水时，:obj:`Default.NoneValue` 。当tables存在有-1 或 None值，则返回 :obj:`Default.MissValue`。
    """
    ra = calc_ra(tables)
    rd = calc_rd(tables)
    if ra != Default.MissValue and rd != Default.MissValue:
        if sum(tables) - ra - rd:
            return (tables.tp + tables.tn - ra - rd) / (sum(tables) - ra - rd)
        return Default.NoneValue
    return Default.MissValue

    def calc(cls, truth: np.ndarray, test: np.ndarray, **kwargs: Any) -> None:
        """算法人员调用进行相关评分计算

        Args:
            truth: 真实值数组
            test: 待测试值数组

        Keyword Args:
            threshold(list): 默认为 ``[0, 0.1, 2.5, 8.0, 16.0, 150]``。 分类阈值表，阈值列表最小值和最大值以外的值，认为是异常值，不纳入计算。
            level(list[int]): 默认为 ``[0, 1, ... len(threshold) - 1]``。 需要计算评分的等级。
            names(list[str]): 默认为 ``["TS", "PO", "FAR", "POD", "ACC", "HSS"]``。需要计算的评分内容。
            method(:obj:`str`, optional): 默认为 ``binary``。参数可选择 ``binary`` (二分类), ``multi`` (多分类) 和 ``grade`` (多级)。
            n(int): 默认为 ``0``。领域匹配半径格点数目。
            acc(int): 默认为 ``4``。计算结果的精度值，即保留有效小数点位数
            save_path(str): 默认为 ``./scores.csv``。评分结果csv文件的保存路径。
        """
        res = cls.scores(truth=truth, test=test, **kwargs)
        
        threshold = kwargs.get("threshold", Default.Threshold["rain"]["1h"])
        level = kwargs.get("level", list(range(len(threshold) - 1)))
        score_names = kwargs.get("names", ["TS", "PO", "FAR", "POD", "ACC", "HSS"])
        acc = kwargs.get("acc", 4)
        save_path = kwargs.get("save_path", "./scores.csv")
        
        with open(save_path, mode="w+") as f:
            f.write(",".join(["level"] + score_names) + "\n")
            print("level\t" + "\t".join(score_names))
            for lvl in level:
                tmp = list()
                for name in score_names:
                    tmp_data = res[f"{lvl}"][name]
                    tmp.append(f"{tmp_data:.{acc}f}")
                f.write(",".join([f"{threshold[lvl]}"] + tmp) + "\n")
                print(f"≥ {threshold[lvl]}\t" + "\t".join(tmp))
        print(f'>>>Scores Calculated Done! Result has been saved in "{save_path}"!')

    def classify(cls, truth: np.ndarray, test: np.ndarray, level: int or float, exist_flag: np.ndarray = None,
                 method: str = "binary", n: int = 0) -> ContingencyTable:
        """基于联列表讲数据分为 :obj:`tp`、:obj:`fp`、:obj:`fn` 和 :obj:`tn`.

        .. note::
            TP-命中   FP-误报/空报   FN-漏报   D-否定正确

            二分类 binary method
                ==================  ===================    ===================
                  观测  \\  预测       FCT >= threshold       FCT <  threshold
                ==================  ===================    ===================
                 OBS >= threshold         TP                      FN
                 OBS <  threshold         FP                      TN
                ==================  ===================    ===================

            多分类 multi method
                ==================  ===================    ===================
                  观测  \\  预测        FCT == threshold       FCT =! threshold
                ==================  ===================    ===================
                 OBS == threshold         TP                      FN
                 OBS =! threshold         FP                      TN
                ==================  ===================    ===================

            多级别 grade method
                ==================  ===================    ===================
                  观测  \\  预测        FCT == threshold       FCT <  threshold
                ==================  ===================    ===================
                 OBS == threshold         TP                      FN
                 OBS <  threshold         FP                      TN
                ==================  ===================    ===================

        Args:
            truth: 真实值数组 或 转变数组
            test: 待测试值数组 或 转变数组
            level: 分类等级
            exist_flag: 默认为 ``None`` 。 数据存在与否的标记，shape同truth和test。数组值为True时, 表明truth和test均存在，否则存在缺失。
            method: 默认为 ``binary`` 。参数可选择 ``binary`` (二分类), ``multi`` (多分类) 和 ``grade`` (多级)。
            n: 默认为 ``0`` 。领域匹配半径格点数目。

        Returns:
            联列表分类结果。TP, FP, FN, TN

        """
        # 默认降水 0 ~ 999 mm 区间内
        if not isinstance(exist_flag, np.ndarray):
            exist_flag = np.where((0 <= truth) & (truth < 999) & (0 <= test) & (test < 999), True, False)
        #  领域匹配
        if n > 0:
            assert isinstance(n, int), TypeError("领域匹配半径必须为整型！")
            assert truth.ndim == test.ndim == 2, ValueError("领域匹配仅支持二维数组")
            truth = cls.expand(truth, exist_flag=exist_flag, n=n)
            test = cls.expand(test, exist_flag=exist_flag, n=n)
        if np.sum(exist_flag):
            if method == "binary":
                a = (level <= truth) & exist_flag
                b = (level > truth) & exist_flag
                c = (level <= test) & exist_flag
                d = (level > test) & exist_flag
            elif method == "multi":
                a = (level == truth) & exist_flag
                b = (level != truth) & exist_flag
                c = (level == test) & exist_flag
                d = (level != test) & exist_flag
            elif method == "grade":
                a = (level == truth) & exist_flag
                b = (level > truth) & exist_flag
                c = (level == test) & exist_flag
                d = (level > test) & exist_flag
            else:
                raise KeyError("The classification method only supports 'binary', 'multi' and 'grade'!")
            return ContingencyTable(tp=np.sum(a & c), fp=np.sum(b & c), fn=np.sum(a & d), tn=np.sum(b & d))
        else:
            return ContingencyTable(tp=-1, fp=-1, fn=-1, tn=-1)

    def expand(cls, data: np.ndarray, exist_flag: np.ndarray = None, n: int = 5) -> np.ndarray:
        """模糊空间领域匹配。当前仅支持领域最大值的模糊空间匹配。

        Args:
            data : 待模糊空间领域匹配的数组
            exist_flag : 默认为 ``None``。 数据存在与否的标记，shape同truth和test。数组值为True时, 表明truth和test均存在，否则存在缺失。
            n : 默认为 ``5``。领域匹配的半径，为格点数而非距离或经纬度的度数。

        Returns:
            模糊空间领域匹配的结果。其 shape 与 `data` 相同。
        """
        if exist_flag is None:
            exist_flag = np.where(data < 999, True, False)
        data[~exist_flag] = -1
        data_copy = np.zeros((2 * n, data.shape[0] + 2 * n, data.shape[1] + 2 * n))
        for i in range(2 * n):
            data_copy[i, i: i + data.shape[0], i: i + data.shape[1]] += data
        data_copy[data_copy == -1] = Default.MissValue
        data = np.max(data_copy[:, n: n + data.shape[0], n: n + data.shape[1]], axis=0)
        return data

[文档]    @classmethod
    def scores(cls, truth, test, **kwargs) -> dict:
        """更灵活的评分调用方式，类似calc。但没有封装结果保存与打印。

        Args:
            truth(ndarray): 真实值数组

            test(ndarray): 待测试值数组

        Keyword Args:
            threshold(list): 默认为 ``[0, 0.1, 2.5, 8.0, 16.0, 150]``。 分类阈值表，阈值列表最小值和最大值以外的值，认为是异常值，不纳入计算。
            level(list[int]): 默认为 ``[0, 1, ... len(threshold) - 1]``。 需要计算评分的等级。
            names(list[str]): 默认为 ``["TS", "PO", "FAR", "POD", "ACC", "HSS"]``。需要计算的评分内容。
            method(:obj:`str`, optional): 默认为 ``binary``。参数可选择 ``binary`` (二分类), ``multi`` (多分类) 和 ``grade`` (多级)。
            n(int): 默认为 ``0``。领域匹配半径格点数目。
            acc(int): 默认为 ``4``。计算结果的精度值，即保留有效小数点位数

        Returns:
            各等级的不同评分结果。
        """
        threshold = kwargs.get("threshold", Default.Threshold["rain"]["1h"])
        # 根据阈值分类原始数据
        truth = cls.transform(data=truth, threshold=threshold)
        test = cls.transform(data=test, threshold=threshold)
        exist_flag = kwargs.get("exist_flag",
                                np.where((truth < len(threshold)) & (test < len(threshold)), True, False))
        # 多指标、多等级评分参数
        level = kwargs.get("level", list(range(len(threshold) - 1)))
        if isinstance(level, int):
            level = [level]
        
        score_names = kwargs.get("names", ["TS", "PO", "FAR", "POD", "ACC", "HSS"])
        method = kwargs.get("method", "binary")
        n = kwargs.get("n", 0)
        acc = kwargs.get("acc", 4)
        
        res = dict()
        for lvl in level:
            tables = cls.classify(truth=truth, test=test, exist_flag=exist_flag, level=lvl, method=method, n=n)
            tmp = dict()
            for name in score_names:
                tmp[name] = np.round(getattr(cls, name.lower())(tables), acc)
            res[f"{lvl}"] = tmp.copy()
        return res
