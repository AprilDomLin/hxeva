    def classify(cls, truth: np.ndarray, test: np.ndarray, exist_flag: Optional[np.ndarray] = None,
                 num: int = 8) -> np.ndarray:
        """ 风向评分分类

        .. note::
        
            以8风向评分为例, 其 num = 8, offset = 22.5°
            
            ==========================  ============================  =============  ==============  ===============  ===============  ===============   ===============   ===============
               观测值  \\   预报值         0.0°~22.5° / 337.5°~360.0°    22.5°~67.5°    67.5°~112.5°    112.5°~157.5°    157.5°~202.5°    202.5°~247.5°     247.5°~292.5°     292.5°~337.5°
            ==========================  ============================  =============  ==============  ===============  ===============  ===============   ===============   ===============
            0.0°~22.5° / 337.5°~360.0°              1                     0.6            0                0                 0                 0                 0               0.6
                  22.5~67.5°                       0.6                     1            0.6               0                 0                 0                 0                0
                  67.5~112.5°                       0                     0.6            1               0.6                0                 0                 0                0
                 112.5~157.5°                       0                      0            0.6               1                0.6                0                 0                0
                 157.5~202.5°                       0                      0             0               0.6                1                0.6                0                0
                 202.5~247.5°                       0                      0             0                0                0.6                1                0.6               0
                 247.5~292.5°                       0                      0             0                0                 0                0.6                1               0.6
                 292.5~337.5°                      0.6                     0             0                0                 0                 0                0.6               1
            ==========================  ============================  =============  ==============  ===============  ===============  ===============   ===============   ===============

        Args:
            truth : 真实转变后的风向数组
            test : 待测试转变后的风向数组
            exist_flag : 默认为 ``None``。数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
            num : 默认为 ``8``。 风向个数划分。

        Returns:
            风向评分分类结果
        """
        # 默认风向 0 ~ 360 度 范围
        if exist_flag is None:
            exist_flag = np.where((0 <= truth) & (truth <= 360) & (0 <= test) & (test <= 360), True, False)
        lvl_data = np.ones_like(truth)
        # lvl_data *= np.where(truth == test, 1.0, 1)                # Score 1.0
        # 以初始化默认情况，不用赋值
        # Score 0.6 （num 7 和 num 0 评分也为0.6）
        lvl_data *= np.where(abs(truth - test) == num - 1, 0.6, 1)
        lvl_data *= np.where(abs(truth - test) == 1, 0.6, 1)  # Score 0.6
        lvl_data *= np.where(abs(truth - test) > 1, 0, 1)  # Score 0
        lvl_data[~exist_flag] = np.nan
        return lvl_data

    

[文档]    @classmethod
    def transform(cls, data: np.ndarray, exist_flag: Optional[np.ndarray] = None, threshold: Optional[list] = None,
                  n: int = 8, offset: float = 22.5) -> np.ndarray:
        """风向根据阈值或个数进行划分。
        其中阈值划分同BaseValue.transform方法，个数划分通常为8风向或4风向的划分（主要取决于参数num的选取）

        Args:
            data: 需根据阈值划分级别的风向数据。
            exist_flag: 默认为 ``None``。数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
            threshold: 默认为 ``None``。风向分类阈值表，阈值列表最小值和最大值以外的值，认为是异常值，不纳入计算。
            n: 默认为 ``8``。风向个数划分。
            offset: 默认为 ``22.5``。偏差度数为风向个数划分的一半度数。如：8风向划分，则偏差度数为 360 / (2 * 8) = 22.5°

        Returns:
            等级划分的风向数据(cdata)
        """
        if isinstance(threshold, (list, np.ndarray)) and len(threshold) > 1:
            lvl_data = BaseEvaluate.transform(data, threshold, exist_flag)
        else:
            # 默认风向 0 ~ 360 度 范围
            if not isinstance(exist_flag, np.ndarray):
                exist_flag = np.where((0 <= data) & (data <= 360), True, False)
            assert 360 / (2 * n) == offset, ValueError(
                f"Expected got {360 / (2 * n)}, but got 'offset' {offset}!")
            lvl_data = data + offset
            lvl_data[data <= offset] += 360
            lvl_data = lvl_data // (360 / n) - 1
        lvl_data[~exist_flag] = np.nan
        return lvl_data

    def severer(cls, truth: np.ndarray, test: np.ndarray, exist_flag: Optional[np.ndarray] = None) -> float:
        """风速预报偏强率(Forecast Severer WindSpeed, FSs)
        
        :math:`severer_i = \\begin{cases} 1, & ctest_i > ctruth_i  \\\\  0, & ctest_i \\leq ctruth_i \\end{cases}`
        
        :math:`FSs = \\frac{1}{N} * \\sum{severer_i}`
        
        Args:
            truth : 真实转变后的风速数组
            test : 待测试转变后的风速数组
            exist_flag : 默认为 ``None``。数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。

        Returns:
            风速预报偏弱率
        """
        # 默认风速在 0级 ~ 14级 风速区间内
        if exist_flag is None:
            exist_flag = np.where((0 <= truth) & (truth < 999) & (0 <= test) & (test < 999), True, False)
        tmp = np.where(truth < test, 1, 0)
        if np.sum(exist_flag):
            return float(np.nanmean(tmp[exist_flag]))
        return np.nan

[文档]    @classmethod
    def accuracy(cls, cdata: np.ndarray, exist_flag: Optional[np.ndarray] = None) -> float:
        """准确率(评分转化为，仅有0和1)
        
        :math:`accuracy_i = \\begin{cases}1 , & truth = test \\\\ 0, & truth \\neq test \\end{cases}`
        
        Args:
            cdata : 风速/风向分类后的数据(即_classify 处理后的数据)矩阵
            exist_flag : 默认为 ``None``。数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。

        Returns:
              风速/风向准确率
        """
        # 默认风速/风向准确率结果不为 np.nan 即为存在
        if exist_flag is None:
            exist_flag = np.where(cdata != np.nan, True, False)
        tmp = np.where((0 <= cdata) & (cdata < 1), True, False)
        cdata[tmp] = 0
        if np.sum(exist_flag):
            return float(np.mean(cdata[exist_flag]))
        return np.nan

    def ACd(cls, truth: np.ndarray, test: np.ndarray, exist_flag: Optional[np.ndarray] = None,
            threshold: Optional[list] = None, n: int = 8, offset: float = 22.5) -> float:
        """风向预报准确率
        
        :math:`ACd = \\frac{1}{N} * \\sum{ACd_i}`
        
        Args:
            truth : 真值风向数据矩阵
            test :  测试值风向数据矩阵
            exist_flag : 默认为 ``None``。数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
            threshold: 默认为 ``None``。风向分类阈值表，阈值列表最小值和最大值以外的值，认为是异常值，不纳入计算。
            n: 默认为 ``8``。风向个数划分。
            offset: 默认为 ``22.5``。偏差度数为风向个数划分的一半度数。如：8风向划分，则偏差度数为 360 / (2 * 8) = 22.5°

    
        Returns:
            风向预报准确率
        """
        if exist_flag is None:
            exist_flag = np.where((0 <= truth) & (truth <= 360) & (0 <= test) & (test <= 360), True, False)
        if np.sum(exist_flag):
            # 原始风向值数据(°) 预处理
            truth = WindDirection.transform(truth, exist_flag, threshold=threshold, n=n, offset=offset)
            test = WindDirection.transform(test, exist_flag, threshold=threshold, n=n, offset=offset)
            return cls.accuracy(WindDirection.classify(truth=truth, test=test, exist_flag=exist_flag))
        return np.nan

            threshold : 默认为 ``蒲氏风速表``。分类阈值表，阈值列表最小值和最大值以外的值，认为是异常值，不纳入计算。

        Returns:
            风速预报准确率
        """
        if exist_flag is None:
            exist_flag = np.where((0 <= truth) & (truth < 999) & (0 <= test) & (test < 999), True, False)
        if np.sum(exist_flag):
            # 原始风速值数据(m/s) 预处理
            truth = WindSpeed.transform(truth, exist_flag, threshold=threshold)
            test = WindSpeed.transform(test, exist_flag, threshold=threshold)
            exist_flag *= np.where((cmin <= truth) & (truth <= cmax) & (cmin <= test) & (test <= cmax), True, False)
            if np.sum(exist_flag):
                return cls.accuracy(WindSpeed.classify(truth=truth, test=test, exist_flag=exist_flag))
        return np.nan

    def ACz(cls, ws_truth: np.ndarray, wd_truth: np.ndarray, ws_test: np.ndarray, wd_test: np.ndarray,
            exist_flag: Optional[np.ndarray] = None, **kwargs: Any) -> float:
        """风预报综合准确率
    
        :math:`ACz = \\frac{1}{N} * \\sum{(ACd_i * ACs_i)}`
        
        Args:
            ws_truth : 真值风速大小数据矩阵
            wd_truth : 真值风向大小数据矩阵
            ws_test : 测试值风速大小数据矩阵
            wd_test : 测试值风向大小数据矩阵
            exist_flag : 默认为 ``None``。数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。

        Keyword Args:
            ws_cmin(int): 默认为 ``0``。
            ws_cmax(int): 默认为 ``14``。
            wd_cmin(int): 默认为 ``0``。
            wd_cmax(int): 默认为 ``8``。
            ws_threshold(list): 默认为 ``None``。风速分类阈值表，阈值列表最小值和最大值以外的值，认为是异常值，不纳入计算。
            wd_threshold(list): 默认为 ``None``。风向分类阈值表，阈值列表最小值和最大值以外的值，认为是异常值，不纳入计算。
            num(int): 默认为 ``8``。风向个数划分。
            offset(float): 默认为 ``22.5``。偏差度数为风向个数划分的一半度数。如：8风向划分，则偏差度数为 360 / (2 * 8) = 22.5°

        Returns:
            风预报综合准确率
        """
        ws_cmin = kwargs.pop("ws_cmin", 0)
        ws_cmax = kwargs.pop("ws_cmax", 14)
        wd_cmin = kwargs.pop("wd_cmin", 0)
        wd_cmax = kwargs.pop("wd_cmax", 8)
        ws_thrshold = kwargs.pop("ws_threshold", None)
        wd_threshold = kwargs.pop("wd_threshold", None)
        n = kwargs.pop("n", 8)
        offset = kwargs.pop("offset", 22.5)
        if exist_flag:
            exist_flag = np.where((0 <= ws_truth) & (ws_truth < 999) & (0 <= ws_test) & (ws_test < 999), True, False)
            exist_flag *= np.where((0 <= wd_truth) & (wd_truth <= 360) & (0 <= wd_test) & (wd_test <= 360), True, False)
        if np.sum(exist_flag):
            #  原始风速数据(m/s)预处理
            ws_truth = WindSpeed.transform(ws_truth, exist_flag, threshold=ws_thrshold)
            ws_test = WindSpeed.transform(ws_test, exist_flag, threshold=ws_thrshold)
            #  原始风向数据(°)预处理
            wd_truth = WindDirection.transform(wd_truth, exist_flag, threshold=wd_threshold, n=n, offset=offset)
            wd_test = WindDirection.transform(wd_test, exist_flag, threshold=wd_threshold, n=n, offset=offset)
            
            exist_flag *= np.where((ws_cmin <= ws_truth) & (ws_truth <= ws_cmax) & (ws_cmin <= ws_test)
                                   & (ws_test <= ws_cmax) & (wd_cmin <= wd_truth) & (wd_truth <= wd_cmax)
                                   & (wd_cmin <= wd_test) & (wd_test <= wd_cmax), True, False)
            
            if np.sum(exist_flag):
                cws = WindSpeed.classify(ws_truth, ws_test, exist_flag)
                cwd = WindDirection.classify(wd_truth, wd_truth, exist_flag)
                return float(np.nanmean((cws * cwd)[exist_flag]))
        return np.nan

            test = WindSpeed.transform(test, exist_flag)
            return WindSpeed.weaker(truth, test, exist_flag)

    def winddirection(cls, truth: np.ndarray, test: np.ndarray, exist_flag: Optional[np.ndarray],
                      acc: int = 4, threshold: Optional[list] = None, n: int = 8, offset: float = 22.5) -> dict:

            threshold: 默认为 ``None``。风向分类阈值表，阈值列表最小值和最大值以外的值，认为是异常值，不纳入计算。

        Returns:
            风向评分结果
        """

        if exist_flag is None:
            exist_flag = np.where((0 <= truth) & (truth <= 360) & (0 <= test) & (test <= 360), True, False)
        
        # 原始风向值数据(°) 预处理
        truth = WindDirection.transform(truth, exist_flag=exist_flag, threshold=threshold, n=n, offset=offset)
        test = WindDirection.transform(test, exist_flag=exist_flag, threshold=threshold, n=n, offset=offset)

        truth = WindSpeed.transform(truth, exist_flag=exist_flag, threshold=threshold)
