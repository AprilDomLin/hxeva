[文档]    @classmethod
    def transform(cls, data: np.ndarray, threshold: list,
                  exist_flag: np.ndarray or None = None) -> np.ndarray:
        """根据阈值(threshold)，对连续变量数据划分为离散的分级数值，便于联列表评分计算。

        Args:
            data : 需根据阈值划分级别的数据。
            threshold : 分类阈值表，阈值列表最小值和最大值以外的值，认为是异常值，不纳入计算。
            exist_flag : 默认为``None``。 数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。

        Returns:
            等级划分数据
        """
        if exist_flag is None:
            exist_flag = np.ones_like(data).astype(np.bool_)
        lvl_data = np.ones_like(data) * Default.MissValue
        for lvl in range(len(threshold) - 1):
            tmp = np.where((threshold[lvl] <= data) & (data < threshold[lvl + 1]), True, False)
            lvl_data[tmp] = lvl
        lvl_data[~exist_flag] = Default.MissValue
        return lvl_data
