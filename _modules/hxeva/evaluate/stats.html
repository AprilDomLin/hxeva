def _stats_check(truth, test, exist_flag):
    if exist_flag is None:
        exist_flag = np.where((0 <= truth) & (truth < 999), True, False)
        exist_flag *= np.where((0 <= test) & (test < 999), True, False)
    return exist_flag



[文档]def calc_cc(truth: np.ndarray, test: np.ndarray, exist_flag: Optional[np.ndarray] = None) -> Union[np.ndarray, float]:
    """ 相关系数计算(Correlation Coefficient, CC or R2)
    
    :math:`CC = \\frac{\\sum{(truth_i - \\overline{truth}) * (test_i - \\overline{test})}}{\\sqrt{\\sum{(truth_i - \\overline{truth})^2} * \\sum{(test_i - \\overline{test})^2}}}`

    
    Args:
        truth : 真实值数组
        test : 待测试值数组
        exist_flag: 默认为 ``None``。 数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
        
    Returns:
         相关系数
    """
    exist_flag = _stats_check(truth, test, exist_flag)
    if np.sum(exist_flag):
        truth, test = truth[exist_flag], test[exist_flag]
        mTruth, mTest = np.mean(truth), np.mean(test)
        numerator = np.sum((truth - mTruth) * (test - mTest))
        denominator = (np.sum((truth - mTruth) ** 2) *
                       np.sum((test - mTest) ** 2)) ** 0.5
        cc = numerator / denominator if denominator else np.nan
    else:
        cc = np.nan
    return cc




[文档]def calc_mae(truth: np.ndarray, test: np.ndarray, exist_flag: Optional[np.ndarray] = None,
             mid: bool = False) -> Union[np.ndarray, float]:
    """ 平均绝对误差(Mean Absolute Error, MAE)
    
    :math:`MAE = \\frac{1}{N} * \\sum{\\left|{truth_i - test_i}\\right|}`
    
    Args:
        truth : 真实值数组
        test : 待测试值数组
        exist_flag: 默认为 ``None``。 数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
        mid: 默认为 ``False``。 计算时，保存格点中间值，而不完成所有格点的均值计算。
        
    Returns:
        平均绝对误差
    """
    exist_flag = _stats_check(truth, test, exist_flag)
    mae = abs(truth - test)
    if mid:
        mae[~exist_flag] = np.nan
    elif np.sum(exist_flag):
        mae = np.mean(mae[exist_flag], dtype=np.float64)
    else:
        mae = np.nan
    return mae




[文档]def calc_mape(truth: np.ndarray, test: np.ndarray, exist_flag: Optional[np.ndarray] = None,
              mid: bool = False) -> Union[np.ndarray, float]:
    """ 平均绝对误差百分比(Mean Absolute Percentage Error, MAPE)
    
    :math:`MAPE = \\frac{\\frac{1}{N} * \\sum{\\left|{truth_i - test_i}\\right|}}{\\overline{truth}}`
    
    Args:
        truth : 真实值数组
        test : 待测试值数组
        exist_flag: 默认为 ``None``。 数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
        mid: 默认为 ``False``。 计算时，保存格点中间值，而不完成所有格点的均值计算。
        
    Returns:
       平均绝对误差百分比
    """
    exist_flag = _stats_check(truth, test, exist_flag)
    if mid:
        tmp_flag = np.where(truth == 0, False, True)
        mape = np.divide(np.abs(test - truth), truth, where=tmp_flag) * tmp_flag
        mape[~(exist_flag & ~tmp_flag)] = np.nan
    elif np.sum(exist_flag):
        mape = np.abs(test - truth)
        mean_truth = np.mean(truth[exist_flag])
        if mean_truth != 0:
            mape = np.mean(mape[exist_flag], dtype=np.float64) / mean_truth
        else:
            mape = np.nan
    else:
        mape = np.nan
    return mape

    """
    exist_flag = _stats_check(truth, test, exist_flag)
    if mid:
        tmp_flag = np.where(truth == 0, False, True)
        mbr = np.divide(test, truth, where=tmp_flag) * tmp_flag
        mbr[~(exist_flag & ~tmp_flag)] = np.nan
    elif np.sum(exist_flag):
        mbr = test
        mean_truth = np.mean(truth[exist_flag])
        if mean_truth != 0:
            mbr = np.mean(mbr[exist_flag], dtype=np.float64) / mean_truth
        else:
            mbr = np.nan

        mbr = np.nan
    return mbr



def calc_nme(truth: np.ndarray, test: np.ndarray, exist_flag: Optional[np.ndarray] = None,
             mid: bool = False) -> Union[np.ndarray, float]:
    """规范化平均偏差(Normalized Mean Error, NME)
    
    :math:`NME = \\frac{\\frac{1}{N} * \\sum{(test_i - truth_i)}}{\\overline{truth}}`
    
    Args:
        truth : 真实值数组
        test : 待测试值数组
        exist_flag: 默认为 ``None``。 数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
        mid: 默认为 ``False``。 计算时，保存格点中间值，而不完成所有格点的均值计算。
        
    Returns:
        规范化平均偏差
    """
    exist_flag = _stats_check(truth, test, exist_flag)
    if mid:
        tmp_flag = np.where(truth + test == 0, False, True)
        nme = np.divide((test - truth), (truth + test), where=tmp_flag) * tmp_flag
        nme[~(exist_flag & ~tmp_flag)] = np.nan
    elif np.sum(exist_flag):
        mean_truth = np.mean((truth + test)[exist_flag])
        if mean_truth != 0:
            nme = np.mean((test - truth)[exist_flag]) / mean_truth
        else:
            nme = np.nan
    else:
        nme = np.nan
    return nme

[文档]def calc_nse(truth: np.ndarray, test: np.ndarray, exist_flag: Optional[np.ndarray] = None,
             mid: bool = False, limit: Union[int, float] = -999) -> Union[np.ndarray, float]:
    """纳什效率系数(Nash–Sutcliffe efficiency score, NSE)

    :math:`NSE = 1 - \\frac{\\sum(truth_i - test_i)^2}{\\sum{(truth_i - \\overline{truth})^2}}`

    Args:
        truth : 真实值数组
        test : 待测试值数组
        exist_flag: 默认为 ``None``。 数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
        mid: 默认为 ``False``。 计算时，保存格点中间值，而不完成所有格点的均值计算。
        limit: 默认为 ``-999``。 NSE值最小值限制，防止数值过低导致溢出。
        
    Returns:
        纳什效率系数
    """
    exist_flag = _stats_check(truth, test, exist_flag)
    if mid:
        raise KeyError("Correlation Coefficient cannot be calculated intermediately")
    elif np.sum(exist_flag):
        truth, test = truth[exist_flag], test[exist_flag]
        denominator = np.sum((truth - np.mean(truth)) ** 2)
        if denominator:
            nse = (1 - np.sum((truth - test) ** 2) / denominator)
            nse = max(limit, nse)
        else:
            nse = np.nan
    else:
        nse = np.nan
    return nse




[文档]def calc_rmse(truth: np.ndarray, test: np.ndarray, exist_flag: Optional[np.ndarray] = None,
              mid: bool = False) -> Union[np.ndarray, float]:
    """均方根误差计算(Root-Mean-Square Error, RMSE)

    :math:`RMSE = \\sqrt{\\frac{1}{N} * \\sum{(truth_i - test_i) ^ 2}}`

    Args:
        truth : 真实值数组
        test : 待测试值数组
        exist_flag: 默认为 ``None``。 数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
        mid: 默认为 ``False``。 计算时，保存格点中间值，而不完成所有格点的均值计算。
        
    Returns:
        均方根误差
    """
    exist_flag = _stats_check(truth, test, exist_flag)
    rmse = (truth - test) ** 2
    if mid:
        rmse[~exist_flag] = np.nan
    elif np.sum(exist_flag):
        rmse = np.mean(rmse[exist_flag]) ** 0.5
    else:
        rmse = np.nan
    return rmse

[文档]    @classmethod
    def calc(cls, truth: np.ndarray, test: np.ndarray, **kwargs: Any) -> None:
        """算法人员调用进行相关统计误差计算

        Args:
            truth : 真实值数组
            test : 待测试值数组

        Keyword Args:
            exist_flag(:obj:`ndarray`) : 默认为 ``None``。数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
            amin(int or float) : 默认为 ``0``。数据最小边界值。
            amax(int or float) : 默认为 ``999``。数据最大边界值。
            names(list[str]) : 默认为 ``["RMSE", "MAE", "MBE", "MAPE", "CC"]``。需要计算的统计误差指标。
            acc(int): 默认为 ``4``。计算结果的精度值，即保留有效小数点位数
            save_path(str) : 默认为 ``./stats.csv``。评分结果csv文件的保存路径。

        Returns:
            统计误差结果
        """
        amin = kwargs.pop("amin", 0)
        amax = kwargs.pop("amax", 999)
        exist_flag = kwargs.pop("exist_flag",
                                np.where((amin <= truth) & (truth < amax) & (amin <= test) & (test < amax),
                                         True, False))
        names = kwargs.pop("names", ["RMSE", "MAE", "MBE", "MAPE", "CC"])
        acc = kwargs.pop("acc", 4)
        save_path = kwargs.pop("save_path", "./stats.csv")
        
        # 多指标统计计算
        res = cls.stats(truth=truth, test=test, amin=amin, amax=amax, exist_flag=exist_flag, names=names, acc=acc)
        
        with open(save_path, mode="w+") as f:
            for name in names:
                tmp = res[name]
                print(tmp)
                f.write(f"{tmp}\n")
        print(f'>>>Statistics Calculated Done! Result has been saved in "{save_path}"!')

            exist_flag : 数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。

        Returns:
            均方根误差
        """
        return calc_rmse(truth, test, exist_flag=exist_flag, mid=False)

    

[文档]    @classmethod
    def stats(cls, truth: np.ndarray, test: np.ndarray, **kwargs: Any) -> dict:
        """更灵活的统计误差调用方式，类似calc。但没有封装结果保存与打印。

        Args:
            truth : 真实值数组
            test : 待测试值数组

        Keyword Args:
            exist_flag(:obj:`ndarray`) : 默认为 ``None``。数据存在与否的标记，shape同truth和test。数组值为True时,表明truth和test均存在，否则存在缺失。
            amin(int or float) : 默认为 ``0``。数据最小边界值。
            amax(int or float) : 默认为 ``999``。数据最大边界值。
            names(list[str]) : 默认为 ``["RMSE", "MAE", "MBE", "MAPE", "CC"]``。需要计算的统计误差指标。
            acc(int): 默认为 ``4``。计算结果的精度值，即保留有效小数点位数

        Returns:
            统计误差结果
        """
        
        # 根据阈值分类原始数据
        amin = kwargs.pop("amin", 0)
        amax = kwargs.pop("amax", 999)
        exist_flag = kwargs.pop("exist_flag",
                                np.where((amin <= truth) & (truth < amax) & (amin <= test) & (test < amax),
                                         True, False))
        
        # 多指标、多等级评分参数
        names = kwargs.pop("names", ["RMSE", "MAE", "MBE", "MAPE", "CC"])
        acc = kwargs.pop("acc", 4)
        
        res = dict()
        for name in names:
            res[name] = np.round(getattr(cls, name.lower())(truth, test, exist_flag), acc)
        return res
