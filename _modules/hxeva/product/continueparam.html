"""
@file       : hxeva
@Function   : 模式---连续要素的检验
              地面连续要素:
                    温度: t2
                    露点温度: dpt
                    相对湿度: rhu
                    气压: plev
                    风速: WIN_S_Avg_10mi
                    风向: WIN_D_Avg_10mi
@Author     : Suruiheng
@Date       : 2022/12/30   16:17
            : 2023/01/18          修改并添加docstring。 --fang linwanghao
@Tool       : PyCharm
@File       : continueparam.py
"""

from pathlib import Path
from typing import Any
from datetime import datetime

import numpy as np
import pandas as pd

from .base import BaseProdEvaluate, DataReader, Log
from ..config import Default, General
from ..evaluate import Stats

__all__ = ["ContinueParamGrid"]



[文档]class ContinueParamGrid(BaseProdEvaluate):
    """ 连续要素产品格点检验评估
    
    Args:
        name : 产品名称。
        log_root : 默认为 ``./logs``。日志文件夹根目录
        extend : 经纬度范围。
    """
    
    def __init__(self, name: str, log_root: str = "./logs", extend: dict or None = None):
        super(ContinueParamGrid, self).__init__(name=name, log_root=log_root)
        if isinstance(extend, dict):
            # 根据extend 更新 General 参数
            General.SLAT = extend["slat"]
            General.SLON = extend["slon"]
            General.ELAT = extend["elat"]
            General.ELON = extend["elon"]
    

[文档]    def evaluate(self, tag_time: datetime, truth_file: dict, test_file: dict, **kwargs: Any) -> bool:
        """ 连续要素产品格点检验评估
        
        Args:
            tag_time : 当前时间
            truth_file : 真值数据信息。参数信息关键字参考CMAE config。
            test_file : 待测数据信息。参数信息关键字参考CMAE config。
            
        Keyword Args:
            surface(bool): 是否为地面连续要素，True: 地面连续要素。False: 高空连续要素，含高度层信息。
            file(str): 检验评估产品保存路径规则。
            timefcst(list): 预报时长表。涵盖预报产品所有的预报时长的列表。
            keys(list): 默认为 ``["t2", "rhu", "win_spd", "win_dir"]``。检验评估的变量关键字列表。
            location(dict or None): 默认为 ``None``。检验评估经纬度范围。
            code(str): 默认为 ``EC``。检验评估对象的名称或代号。
            max_counter(int): 默认为 ``10``。 数据产品检验评估错误时，循环等待的次数。
            
        Returns:
            是否正常完成评估。True 正常完成评估; False 未正常完成评估，或因路径错误。
        """
        # Keyword Args
        surface = kwargs.pop("surface")
        save = kwargs.pop("file")
        timefcst = kwargs.pop("timefcst")
        keys = kwargs.get("keys", ["t2", "rhu", "win_spd", "win_dir"])
        loc = kwargs.get("location", None)
        code = kwargs.get("code", "EC")
        max_counter = kwargs.get("max_counter", 10)
        # 预处理
        # 更新General 经纬度信息参数
        if loc is not None and isinstance(loc, dict):
            for _ in ["slat", "slon", "elat", "elon"]:
                setattr(General, _.upper(), loc.pop(_))
        # 处理地面/高空的关键信息
        if surface:
            upar = dict()
        else:
            upar = dict(surface=surface, name_Height=kwargs.get("name_Height", "Plev"))
            # 高空产品检验评估时，必须保证General PLEV参数为非空
            if General.PLEV is None:
                setattr(General, "PLEV", loc.pop("plev"))
        
        for lead_time in timefcst:
            fcst_time = tag_time + lead_time
            # 读取truth数据    >>> UTC
            path = Path(DataReader.catch(time=self.transform_time(fcst_time, timezone=truth_file["timezone"]),
                                         lead_time=None, file=truth_file["file"], ))
            if path.exists():
                Log.info(self.logger, f"Success catch the truth Data: {path}")
            elif self.counter <= max_counter:
                Log.warning(self.logger, f"The truth Data is missing, {path}!!!")
                return False
            else:
                Log.warning(self.logger, f"The truth Data is missing, {path}!!!   !!!And go on evaluate...!!!")
            truth = DataReader.load(path, vnames=keys, name_Lat=truth_file["name_Lat"],
                                    name_Lon=truth_file["name_Lon"], surface=surface, **upar)
            
            # 读取test数据     >>> UTC
            path = Path(DataReader.catch(time=self.transform_time(tag_time, timezone=test_file["timezone"]),
                                         lead_time=lead_time, file=test_file["file"], ))
            if path.exists():
                Log.info(self.logger, f"Success catch the forecast product: {path}")
            elif self.counter <= max_counter:
                Log.warning(self.logger, f"The forecast product is missing, {path}!!!")
                return False
            else:
                Log.warning(self.logger, f"The forecast product is missing, {path}!!!   !!!And go on evaluate...!!!")
            test = DataReader.load(path, vnames=keys, name_Lat=test_file["name_Lat"],
                                   name_Lon=test_file["name_Lon"], surface=surface, **upar)
            
            # 检验评估
            res = pd.DataFrame(columns=["area", "mode", "stime", "ftime"])  # 初始化检验评估结果
            names = ["RMSE", "MAE", "MBE"]
            for key in keys:
                if surface:
                    # 地面连续要素检验
                    idx = len(res)
                    exist_flag = np.where((truth[key] == Default.MissValue) & (test[key] == Default.MissValue),
                                          False, True)
                    for name in names:
                        res.loc[idx, f"{key}_{name}"] = Default.MissValue
                    
                    tmp = Stats.stats(truth=truth[key], test=test[key], exist_flag=exist_flag, names=names)
                    for name in names:
                        res[name] = tmp[name]
                else:
                    # 高空连续要素检验，循环气压高度层
                    for plev in General.PLEV:
                        idx = len(res)
                        exist_flag = np.where(
                            (truth[key][plev] == Default.MissValue) & (test[key][plev] == Default.MissValue), False,
                            True)
                        for name in names:
                            res.loc[idx, f"{key}_{name}"] = Default.MissValue
                        
                        tmp = Stats.stats(truth=truth[key][plev], test=test[key][plev], exist_flag=exist_flag,
                                          names=names)
                        for name in names:
                            res[name] = tmp[name]
                        res.loc[idx, "plev"] = plev
            
            # 补全相关信息
            res.loc[:, "area"] = kwargs.get("area", "全国")
            res.loc[:, "mode"] = code
            res.loc[:, "stime"] = tag_time.strftime("%Y%m%d%H")
            res.loc[:, "ftime"] = fcst_time.strftime("%Y%m%d%H")
            
            # 保存结果
            save_path = Path(DataReader.merge_file(time=tag_time, lead_time=lead_time, file=save))
            save_path.parent.mkdir(parents=True, exist_ok=True)
            sep = "\t" if save_path.suffix == "txt" else ","
            res.to_csv(save_path, sep=sep, index_label=False)
