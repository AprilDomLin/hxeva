[文档]class HeavyRainFcstSta(BaseProdEvaluate, ABC):
    """ 强降水概率预报产品站点检验评估

    Args:
        name : 默认为 ``HeavyRain``。产品名称。
        log_root : 默认为 ``./logs``。日志文件夹根目录
        max_counter: 默认为 ``10``。文件错误最大循环等待次数
        shape: 默认为 ``None``。评估数据形状大小
        extend : 默认为 ``None``。经纬度范围。
    """

    def __init__(self, name: str = "HeavyRain", log_root: str = "./logs", max_counter: int = 10,
                 shape: Optional[tuple] = None, extend: Optional[dict] = None) -> None:
        super(HeavyRainFcstSta, self).__init__(name=name, log_root=log_root, max_counter=max_counter, shape=shape)
        self.extend = extend
    

[文档]    def evaluate(self, tag_time: datetime, truth_file: dict, test_file: dict, **kwargs: Any) -> bool:
        """ 强降水概率预报产品站点检验评估
        
        Args:
            tag_time : 当前时间
            truth_file : 真值数据信息。参数信息关键字参考CMAE config。
            test_file : 待测数据信息。参数信息关键字参考CMAE config。
        
        Keyword Args:
            save(str): 检验评估产品保存路径规则。
            timefcst(list): 预报时长表。涵盖预报产品所有的预报时长的列表。
            key(list): 默认为 ``HeavyRain``。检验评估的变量关键字。
            threshold(list): 默认为 ``[0, 50, 101]``。强降水概率预报阈值表。
            location(dict or None): 默认为 ``None``。检验评估经纬度范围。
            code(str): 默认为 ``HEAVYRAIN_03H_STATION``。检验评估对象的名称或代号。
            area(str): 默认为 ``None``。 检验区域名称，检验评估产品字段。

        Returns:
            是否正常完成评估。True 正常完成评估; False 未正常完成评估，或因路径错误。
        """
        # Keyword Args
        save = kwargs.pop("save")
        timefcst = kwargs.pop("timefcst")
        key = kwargs.get("key", "HeavyRain")
        threshold = kwargs.get("threshold", [0, 50, 101])
        loc = kwargs.get("location", None)
        code = kwargs.get("code", "HEAVYRAIN_03H_STATION")
        area = kwargs.get("area", None)
        
        stainfo = ["Station_Name", "Province", "City", "Cnty", "Station_Id_C",
                   truth_file["name_Lat"], truth_file["name_Lon"]]
        stavars = ["PRE_1h"]
        # 读取 Truth
        path = Path(DataReader.catch(tag_time, lead_time=None, file=truth_file["file"]))
        if not self._path_judge(path, "PRE Station(Truth)"):
            return False
        if path:
            truth = pd.read_csv(path, sep='\t', low_memory=False)
            # 提取对应区域站点和目标变量
            flag_lat = (loc["slat"] - 0.1 <= truth.Lat) & (truth.Lat <= loc["elat"] + 0.1)
            flag_lon = (loc["slon"] - 0.1 <= truth.Lon) & (truth.Lon <= loc["elon"] + 0.1)
            truth = truth[flag_lat & flag_lon]
            heavy_flag = (20.0 <= truth.PRE_1h) & (truth.PRE_1h <= 150)
            truth.loc[heavy_flag, key] = 100
            truth.loc[~heavy_flag, key] = 0
        else:
            truth = pd.DataFrame(columns=stainfo + stavars)
        
        for lead_time in timefcst:
            rep_time = tag_time - lead_time
            # 读取 Test
            path = Path(DataReader.catch(rep_time, lead_time=lead_time, file=test_file["file"]))
            if not self._path_judge(path, "Heavy Rain(Test)"):
                return False
            test = DataReader.load(path, vnames=[key], extend=loc)

            # 实况与预报产品格点匹配
            old_lon, old_lat = np.meshgrid(test[test_file["name_Lon"]], test[test_file["name_Lat"]])
            test = self.data_match(test, key=key, old_lat=old_lat, old_lon=old_lon,
                                   new_lat=truth.Lat, new_lon=truth.Lon, window_size=(5, 5), stride=(1, 1))

            # 初始化检验结果
            ts_res = pd.DataFrame(columns=["stime", "ftime", "area", "type", "method", "PO", "POD", "FAR", "TS"])
            ts_res.loc[0, "stime"] = rep_time.strftime("%Y%m%d%H%M")
            ts_res.loc[0, "ftime"] = tag_time.strftime("%Y%m%d%H%M")
            ts_res.loc[0, "area"] = area
            ts_res.loc[0, "type"] = code
            ts_res.loc[0, "method"] = "TS"
            
            # 评估
            ts_path = Path(DataReader.merge_file(time=rep_time, lead_time=lead_time, file=save["ts"]))
            Path(ts_path).parent.mkdir(parents=True, exist_ok=True)
    
            tmp = Score.scores(np.array(truth.HeavyRain), test["HeavyRain"], threshold=threshold,
                               score_names=["PO", "POD", "FAR", "TS"])
            for name in ["PO", "POD", "FAR", "TS"]:
                ts_res.loc[0, name] = 999 if np.isnan(tmp["1"][name]) else tmp["1"][name]
    
            ts_res.to_json(ts_path, orient="records", force_ascii=False, lines=True)
        return True
