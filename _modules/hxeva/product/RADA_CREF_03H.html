from abc import ABC
from pathlib import Path
import json

import numpy as np
from scipy.interpolate import griddata

from .base import BaseProdEvaluate, DataReader, Log
from ..config import Default, General
from ..evaluate import Score, Stats


__all__ = ["RadaCREF03HGrid"]



[文档]class RadaCREF03HGrid(BaseProdEvaluate, ABC):
    def __init__(self, name="RADA_CREF_03H_GRID", log_root="./logs", extend=None):
        super(RadaCREF03HGrid, self).__init__(name=name, log_root=log_root)
        if isinstance(extend, dict):
            General.SLAT = extend["slat"]
            General.SLON = extend["slon"]
            General.ELAT = extend["elat"]
            General.ELON = extend["elon"]

    def evaluate(self, tag_time, **kwargs):
        # Args and Kwargs, same with CMAE
        save = kwargs.pop("file")
        truth_file = kwargs.pop("truth_file")
        test_file = kwargs.pop("test_file")

        timefcst = kwargs.pop("timefcst")

        code = kwargs.get("code", "RADA_CREF_03H")
        key = kwargs.get("key", "CREF")
        threshold = kwargs.get("threshold", [-5, 0, 15, 35, 300])
        loc = kwargs.get("location", None)
        # 预处理
        if loc is not None and isinstance(loc, dict):
            for _ in ["slat", "slon", "elat", "elon"]:
                setattr(General, _.upper(), loc.pop(_))

        # 行政边界
        border = kwargs.pop("border")
        border_list = border["BORDERLIST"]
        index = len(border_list)
        border_list.loc[index, "qxCode"] = "-1"
        border_list.loc[index, "dsName"] = "全区平均"
        border_lon, border_lat = np.meshgrid(border["BORDERLON"], border["BORDERLAT"])

        for lead_time in timefcst:
            fcst_time = tag_time + lead_time
            # Load Truth    >>> UTC
            path = Path(
                DataReader.catch(
                    time=self.transform_time(
                        fcst_time, timezone=truth_file["timezone"]
                    ),
                    lead_time=None,
                    file=truth_file["file"],
                )
            )
            if not path.exists():
                if self.counter <= 10:
                    Log.warning(self.logger, f"CMPAS Data is missing, {path}!!!")
                    return False
                else:
                    Log.warning(
                        self.logger,
                        f"CMPAS Data is missing, {path}!!!   !!!And go on evaluate...!!!",
                    )
            else:
                Log.info(self.logger, f"Success catch the CMPAS Data: {path}")
            truth = DataReader.load(
                path,
                vnames=[key],
                name_Lat=truth_file["name_Lat"],
                name_Lon=truth_file["name_Lon"],
            )

            # Load Test     >>> UTC
            path = Path(
                DataReader.catch(
                    time=self.transform_time(
                        tag_time, timezone=test_file["timezone"]
                    ),
                    lead_time=lead_time,
                    file=test_file["file"],
                )
            )
            if not path.exists():
                if self.counter <= 10:
                    Log.warning(self.logger, f"QPF Product is missing, {path}!!!")
                    return False
                else:
                    Log.warning(
                        self.logger,
                        f"QPF Product is missing, {path}!!!   !!!And go on evaluate...!!!",
                    )
            else:
                Log.info(self.logger, f"Success catch the QPF Product: {path}")
            test = DataReader.load(
                path,
                vnames=[key],
                name_Lat=test_file["name_Lat"],
                name_Lon=test_file["name_Lon"],
            )

            # Evaluate
            # 实况、产品与行政边界格点匹配
            if ("Lon" in truth) and ("Lat" in truth) and (key in truth):
                old_lon, old_lat = np.meshgrid(truth["Lon"], truth["Lat"])
                truth[key] = griddata(
                    (old_lon.ravel(), old_lat.ravel()),
                    truth[key].ravel(),
                    (border_lon, border_lat),
                )
            else:
                truth[key] = np.ones_like(border["BORDERCODE"]) * Default.MissValue

            if ("Lon" in test) and ("Lat" in test) and (key in test):
                old_lon, old_lat = np.meshgrid(test["Lon"], test["Lat"])
                test[key] = griddata(
                    (old_lon.ravel(), old_lat.ravel()),
                    test[key].ravel(),
                    (border_lon, border_lat),
                )
            else:
                test[key] = np.ones_like(border["BORDERCODE"]) * Default.MissValue

            ts_path = Path(
                DataReader.merge_file(
                    time=tag_time, lead_time=lead_time, file=save["ts"]
                )
            )
            with open(ts_path, "w", encoding="utf8") as fp:
                pass
            stats_path = Path(
                DataReader.merge_file(
                    time=tag_time, lead_time=lead_time, file=save["stats"]
                )
            )
            with open(stats_path, "w", encoding="utf8") as fp:
                pass

            for col in border_list.itertuples():
                if col.qxCode != "-1":
                    exist_flag = np.where(
                        border["BORDERCODE"] == col.qxCode, True, False
                    )
                else:
                    exist_flag = np.ones_like(border["BORDERCODE"]).astype(np.bool_)

                # TS 评分
                names = ["TS", "PO", "POD", "FAR"]
                # init result
                exist_flag &= np.where(
                    (truth["PRE_1h"] == Default.MissValue)
                    & (test["PRE_1h"] == Default.MissValue),
                    False,
                    True,
                )
                res = dict(
                    stime=tag_time.strftime("%Y%m%d%H%M"),
                    ftime=fcst_time.strftime("%Y%m%d%H%M"),
                    qxName=col.dsName,
                    area=kwargs.get("area"),
                    type=code,
                    method="TS",
                )
                for lvl in Default.RainLevel.values():
                    for name in names:
                        res[f"{lvl}_{name}"] = Default.MissValue
                # evaluate
                tmp = Score.scores(
                    truth=truth["PRE_1h"],
                    test=test["PRE_1h"],
                    exist_flag=exist_flag,
                    threshold=threshold,
                    names=names,
                )
                for key, lvl in Default.RainLevel.items():
                    for name in names:
                        res[f"{lvl}_{name}"] = tmp[key][name]

                with open(ts_path, "a+", encoding="utf8") as fp:
                    fp.write(json.dumps(res, ensure_ascii=False) + "\n")

                # Stats 评分
                names = ["MBR", "MAE", "CC", "NMAE", "NME", "NSE"]
                # init result
                res = dict(
                    stime=tag_time.strftime("%Y%m%d%H%M"),
                    ftime=fcst_time.strftime("%Y%m%d%H%M"),
                    qxName=col.dsName,
                    area=kwargs.get("area"),
                    type=kwargs.get("code"),
                    method="STATISTIC",
                )
                # evaluate
                for name in names:
                    res[name] = Default.MissValue

                tmp = Stats.stats(
                    truth=truth["PRE_1h"],
                    test=test["PRE_1h"],
                    exist_flag=exist_flag,
                    amin=0,
                    amax=150,
                    names=names,
                )
                for name in names:
                    res[name] = tmp[name]

                with open(stats_path, "a+", encoding="utf8") as fp:
                    fp.write(json.dumps(res, ensure_ascii=False) + "\n")
            Log.info(self.logger, f">>>TS Check Result: {ts_path}")
            Log.info(self.logger, f">>>STATISTIC Check Result: {stats_path}")
        return True
