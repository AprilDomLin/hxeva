[文档]class Extend(object):
    """数据空间信息

    包括：起始经度(slon)、结束经度(elon)、起始纬度(slat)、结束纬度(elat)、经纬度精度(error) 和 气压高度层(plev)。

    Attributes:
        slat(float): 起始纬度, start latitude。
        elat(float): 起始径度, start longitude。
        slon(float): 结束纬度, end latitude。
        elon(float): 结束经度, end longitude。
        error(float): 默认为 ``0.05``。经纬度误差。
        plev(list): 默认为 ``None``。气压高度层

    Args:
        slat(float): 起始纬度。
        elat(float): 起始径度。
        slon(float): 结束纬度。
        elon(float): 结束经度。
        error(float): 默认为 ``0.05``。经纬度误差。
        plev(list or float or int): 默认为 ``None``。气压高度层
    """
    
    def __init__(self, slat, elat, slon, elon, error=0.05, plev=None):
        self._slat = slat
        self._elat = elat
        self._slon = slon
        self._elon = elon
        self._error = error
        self._plev = plev
    
    @property
    def slat(self):
        return self._slat - self._error
    
    @property
    def elat(self):
        return self._elat + self._error
    
    @property
    def slon(self):
        return self._slon - self._error
    
    @property
    def elon(self):
        return self._elon + self._error
    
    @property
    def error(self):
        return self._error
    
    @property
    def plev(self):
        if self._plev is None:
            return None
        elif isinstance(self._plev, list):
            return self._plev
        elif isinstance(self._plev, float) or isinstance(self._plev, int):
            return [self._plev]
        else:
            raise ValueError(f'>>>Expected "plev" is the number or list, but got {self._plev}!')

    Attributes:
        name(str): 产品名称。

        logger(logging.Logger): 业务日志记录器。
        counter(int): 默认为 ``0``。计数器。用于业务运行时，判断是否需要忽略错误，生产缺失检验结果。
        
    Args:
        logger : 默认为 ``None``。业务日志记录器。
        name : 默认为 ``空``。产品名称。
        log_root : 默认为 ``./logs``。日志文件夹根目录
    """
    
    def __init__(self, logger: logging.Logger or None = None, name: str = "", log_root: str = "./logs"):
        self.name = name  # 产品名称
        self.log_folder = Path(f"{log_root}/{name}")  # 日志文件夹
        self.log_folder.mkdir(parents=True, exist_ok=True)  # 创建文件夹
        self.logger = logger  # 日志记录器
        self.counter = 0  # 计数器
    

[文档]    def new_log(self, tag_time: str, old_time: str) -> None:
        """ 新建日志
        
        删除 :obj:`old_time` 的旧日志文件，并创建 :obj:`tag_time` 的新日志

        Args:
            tag_time: 当前时间戳
            old_time: 历史待删除时间戳
        """
        if self.logger is None:
            self.logger = logging.getLogger(self.name)
        # 删除old_time 日志
        old_path = self.log_folder.joinpath(f"{old_time}.log")
        old_path.unlink(missing_ok=True)
        # 创建tag_time 日志
        if self.logger.handlers:
            self.logger.handlers.pop()
        self.logger.addHandler(logging.FileHandler(self.log_folder.joinpath(f"{tag_time}.log"), mode="w"))
        self.logger.setLevel(level=logging.INFO)

    

[文档]    def main(self, tag_time: datetime or None = None, **kwargs: Any) -> None:
        """ 实时业务运行主程序
        
        Args:
            tag_time: 当前时间戳。
        
        Keyword Args:
            fmt(str): 默认为 ``%Y%m%d%H``。日志文件保存的时间格式。
            days(int): 默认为 ``7``。 日志保存的时长，单位为: day。
            sleep_time(int): 默认为 ``60``。 当数据出现错误，循环等待的时间，单位为: s。
            delay(timedelta): 默认为 ``20``。 根据tag_time，延迟检验的时间，单位为: min。
        """
        # 从每日00时开始
        if tag_time is None:
            tag_time = datetime.strptime(datetime.now().strftime("%Y%m%d00"), "%Y%m%d%H")
        
        # Keyword Args
        fmt = kwargs.get("fmt", "%Y%m%d%H")                 # 日志文件名格式
        days = kwargs.get("days", 7)                        # 删除历史文件的天数
        sleep_time = kwargs.get("sleep_time", 60)           # 错误运行，休息时间（秒数）
        delay = kwargs.get("delay", timedelta(minutes=20))  # 延迟检验时间
        
        # 需要删除的旧日志
        old_time = tag_time - timedelta(days=days)
        self.new_log(tag_time=tag_time.strftime(fmt), old_time=old_time.strftime(fmt))
        
        st = datetime.now()
        now_time = datetime.now() - delay                  # 需计算数据延迟时间
        Log.begin(self.logger, f"TAG Time : {tag_time}")   # tag_time的业务开始
        if now_time < tag_time:
            # 当前实际时间还未到tag_time时，进入休眠等待
            delta = (tag_time - now_time).total_seconds()
            self.logger.info(f">>>Sleeping...{delta:.1f} s\n")
            sleep(delta)
            
        # 每次新的tag_time，需要重置计数器
        self.counter = 0
        # tag_time的业务循环，至到正常运行检验评估(evaluate)
        while True:
            if self.evaluate(tag_time, **kwargs):
                break
            sleep(sleep_time)
            self.counter += 1
            
        self.logger.info(f">>>Cost: {(datetime.now() - st).total_seconds():.1f} s\n")
        print(f">>>Cost: {(datetime.now() - st).total_seconds():.1f} s")
        Log.end(self.logger, f"{self.name}  --> Done!")   # tag_time的业务结束

    def transform_time(time: datetime, timezone: str, rollback: bool = False) -> datetime:
        """ 时区转化
        
        当前仅支持基于世界时(UTC)的转化和逆转化。
        
        Args:
            time: 时间戳，UTC。
            timezone: 时区名称
            rollback: 默认为 ``False``。是否由 :obj:`timezone` 回滚为世界时。True, 逆转化为UTC。False, 转化为 :obj:`timezone`。

        Returns:
            转化为 :obj:`timezone` 的时间
        """
        flag = -1 if rollback else 1
        if timezone in ["CST", "BJS"]:
            # UTC <-> 北京时
            return time + timedelta(hours=flag * 8)
        elif timezone in ["UTC", "GMT"]:
            # UTC
            return time
        else:
            raise NotImplementedError(f"time zone - {timezone} has not implemented yet!")

[文档]    @classmethod
    def catch(cls, time: datetime, lead_time: timedelta or None, file: str) -> str:
        """ 抓取目标数据路径

        根据时间(nowtime)、预报时效(lead_time)和文件路径规则(file)抓取目标路径

        Args:
            time: 当前时间
            lead_time: 预报时长
            file: 数据路径规则

        Returns:
            目标数据路径

        Exampls:


            >>> from datetime import datetime, timedelta
            >>> tag_time = datetime(year=2022, month=7, day=2, hour=4)   # 目标时间
            >>> file1 = "/mnt/PRESKY/data/cmadata/NAFP/CMPAS_GRIB/%Y/%Y%m/%Y%m%d/*%Y%m%d%H.GRB2"
            >>> file2 = "~/product/FCST/SZW_1000M/F_RADA_FCST_03H_DL/%Y%m%d/F_RADA_FCST_03H_DL_SZW_1000M_%Y%m%d%H%M_%Y%m%d%H%M.nc"
            
            >>> # 抓取实况时间为 2022年7月2日5时，文件路径规则为file1的实况数据
            >>>path1 = DataReader.catch(time=tag_time + timedelta(hours=1), lead_time=None, file=file1)
            >>> # 抓取起报时间为2022年7月2日4时，预报时间为2022年7月2日5时（预报时效为1小时），文件路径规则为file2的预报产品数据
            >>> path2 = DataReader.catch(time=tag_time, lead_time=timedelta(hours=1), file=file2)
        """
        if file.count("*") > 0:
            # 模糊匹配
            if file.count("%Y") + file.count("%m") + file.count("%d") > 0:
                assert lead_time is None, TypeError(f">>>Status: {Status.Abnormal}  -- "
                                                    f"{datetime.now()}\n ftime is exists!")
                file = time.strftime(file)
            path = glob.glob(time.strftime(file))
            path = sorted(path, key=lambda x: x.stat().st_ctime, reverse=True)  # 根据文件修改时间进行排序
            if len(file) > 0:
                path = path[0]  # 返回最新修改时间的模糊匹配文件
            else:
                path = file
        else:
            # 精准匹配
            if len(re.findall("{:0\dd}", file)) == 1 and isinstance(lead_time, timedelta):
                # 匹配模式文件名包含预报时长小时数的文件路径
                file = file.format(int(lead_time.total_seconds() / 3600))
            path = cls.merge_file(time, lead_time=lead_time, file=file)
        return path

    

[文档]    @classmethod
    def load(cls, path: str or Path, vnames: list, **kwargs: Any) -> dict:
        """ 加载格点数据

        根据数据路径(path)，所需读取变量列表(vnames)，及其他参数——包括，经纬度精度(acc)、落区范围(extend)、 grib2变量名称(gribdict)读取目标数据

        Args:
            path : 原始数据/产品路径。
            vnames : 待读取的变量名称列表。

        Keyword Args:
            acc(int): 默认为 ``2``。读取经纬度的保留小数精度。
            extend(dict): 默认为 ``General`` 参数。产品读取经纬度范围，字典由开始经度(slon)、结束经度(elon)、开始纬度(slat)、结束纬度(elat)和经纬度精度(error)组成。
            gribdict(dict): 默认为 ``None``。需补充的GIRB2数据变量选取参数，用于select方法。

        Returns:
            含格点数据的容器

        Examples:

            >>> path1, path2 = "./././path1", "./././path2"
            >>> pre_dict = dict(PRE_1h=dict(parameterName="Total precipitation"))
            >>> cmpas = DataReader.load(path=path1, vnames=["PRE_1h"], gribdict=pre_dict)    # 读取CMPAS实况数据
            >>> qpf = DataReader.load(path=path2, vnames=["PRE_1h"], acc=2)                  # 读取QPF-DL预报产品数据
        """
        # 若数据路径不存在，则返回空容器
        p = Path(path)
        if not p.exists():
            return dict()
        
        error = pow(10, -1 * kwargs.get("acc", 2)) / 2
        if "extend" in kwargs and isinstance(kwargs.get("extend"), dict):
            # 若传入经纬度范围，则更新传入经纬度范围
            extend = Extend(error=error, **kwargs.pop("extend"))
        else:
            # 若未传入经纬度范围，则默认为config.General 的经纬度范围
            extend = Extend(slat=General.SLAT, slon=General.SLON, elat=General.ELAT, elon=General.ELON, error=error, )
        
        # 若基类GRIB2字典表不包含提取变量的关键信息，则传入更新
        if "gribdict" in kwargs and isinstance(kwargs.get("gribdict"), dict):
            cls.GRIBDICT.update(kwargs.pop("gribdict"))
        
        # 选择格点数据读取方式
        ext = p.suffix.lower()
        if ext == ".nc":
            return cls.load_nc(path, vnames, extend, **kwargs)
        elif ext in [".grb2", ".bin"]:
            return cls.load_grb(path, vnames, extend, **kwargs)
        else:
            return dict()

    

[文档]    @classmethod
    def load_nc(cls, path: str or Path, vnames: list, extend: Extend,
                name_Lon: str = "Lon", name_Lat: str = "Lat", **kwargs: Any) -> dict:
        """NC格式数据的读取。

        Args:
            path: 原始数据/产品路径。
            vnames: 待读取的变量名称列表。
            extend: 产品读取经纬度范围，字典由开始经度(slon)、结束经度(elon)、开始纬度(slat)、结束纬度(elat)和经纬度精度(error)组成
            name_Lon: 默认为 ``Lon``。经度坐标NC文件变量名称
            name_Lat: 默认为 ``Lat``。纬度坐标NC文件变量名称

        Keyword Args:
            surface(bool): 是否为地面格点数据
            name_Height(str): 默认为 ``Plev``。高度轴名称。仅当 :obj:`surface` 为 ``True`` 时可用。

        Returns:
            含NC格点数据的容器
        """
        data = Dataset(path)
        # todo 多维NC数据的读取
        x_coord = data.variables[name_Lon][:]
        y_coord = data.variables[name_Lat][:]
        nx = np.where((extend.slon <= x_coord) & (x_coord <= extend.elon))[0]
        ny = np.where((extend.slat <= y_coord) & (y_coord <= extend.elat))[0]
        surface = kwargs.get("surface", True)
        # 当为高空格点数据时的前处理
        if not surface:
            z_coord = data.variables[kwargs.get("name_Height", "Plev")]
            nz = np.where(np.isin(z_coord, extend.plev))
        else:
            nz = None
        # 存放坐标轴
        cntr = {name_Lon: x_coord[nx], name_Lat: y_coord[ny]}
        for vname in vnames:
            if vname in data.variables.keys():
                if surface:
                    cntr[vname] = data.variables[vname][ny, nx]
                else:
                    cntr[vname] = data.variables[vname][nz, ny, nx]
        return cntr

    def load_grb(cls, path: str or Path, vnames: list, extend: Extend,
                 name_Lon: str = "Lon", name_Lat: str = "Lat", **kwargs: Any) -> dict:
        """GRIB格式数据的读取。

        Args:
            path: 原始数据/产品路径。
            vnames: 待读取的变量名称列表。
            extend: 产品读取经纬度范围，字典由开始经度(slon)、结束经度(elon)、开始纬度(slat)、结束纬度(elat)和经纬度精度(error)组成
            name_Lon: 默认为 ``Lon``。经度坐标NC文件变量名称
            name_Lat: 默认为 ``Lat``。纬度坐标NC文件变量名称

        Keyword Args:
            surface(bool): 是否为地面格点数据

        Returns:
            含GRIB格点数据的容器
        """
        surface = kwargs.get("surface", True)
        if not surface:
            # todo GRIB2 格式数据暂不支持3-D 格点读取
            raise NotImplementedError(f"Loading upar GRIB2 data have not implemented yet!")
        cntr = dict()
        with pygrib.open(path) as messages:
            for vname in vnames:
                message = messages.select(**cls.GRIBDICT[vname])
                if len(message) == 1:
                    message = message[0]
                elif len(message) > 1:
                    message = message[0]
                    Warning(f">>> GRIB DATA have more than one message of {vname}!!!")
                else:
                    Warning(f">>> GRIB DATA have no one message of {vname}!!!")
                    continue
                data, lat, lon = message.data(lat1=extend.slat, lon1=extend.slon,
                                              lat2=extend.elat, lon2=extend.elon)
                cntr[vname] = data
                # 当不从数据中读取坐标信息且数据容器中不包含坐标信息时，存放坐标信息
                if (name_Lon not in vnames) and (name_Lon not in cntr):
                    cntr[name_Lon] = lon
                if (name_Lat not in vnames) and (name_Lat not in cntr):
                    cntr[name_Lat] = lat
        return cntr

    

[文档]    @classmethod
    def merge_file(cls, time: datetime, lead_time: timedelta or None, file: str) -> str:
        """组合预测数据的文件名(启报时间、预报时间)

        Args:
            time: 起报时间时间
            lead_time: 预报时长
            file: 数据路径规则

        Returns:
            文件名称
        """
        if lead_time is None:
            # 若预报时长为None，则为实况数据
            return time.strftime(file)
        else:
            # 若预报时长非None，则视time为起报时间
            file_split_1 = time.strftime("_".join(file.split("_")[:-1]))
            file_split_2 = (time + lead_time).strftime(file.split("_")[-1])
            file = file_split_1 + "_" + file_split_2
            return file

    

[文档]    @classmethod
    def read(cls, time: datetime, vnames: list, file: str, **kwargs: Any) -> dict:
        """基于时间、变量名和文件规则读取相应的数据。

        Args:
            time: 目标时间
            vnames: 变量名列表
            file: 数据路径规则

        Keyword Args:
            lead_time(timedelta): 默认为 ``None``。预报时长
            acc(int): 默认为 ``2``。经纬度坐标的读取精度。

        Returns:
            含格点数据的容器
        """
        # 抓取文件路径
        path = cls.catch(time=time, lead_time=kwargs.get("lead_time", None), file=file)
        
        # 边界经纬度
        cntr = cls.load(path=path, vnames=vnames, **kwargs)
        return cntr
