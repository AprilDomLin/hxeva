[文档]    @classmethod
    def read_lpd(cls, path):
        """

        Args:
            path: The path of the LPD lightning data.

        Returns:

        """
        
        def _partStr(string: str):
            tmp = string.split("=")
            if len(tmp):
                return tmp[1]
            return pd.NA
        
        data = pd.read_csv(path, sep="\t", low_memory=False, header=None)
        nums = len(data.columns)
        columns = [None] * nums
        for n in range(nums):
            columns[n] = data.loc[0, n].split("=")[0]
            data.loc[:, n] = data.loc[:, n].apply(_partStr)
        data.columns = columns
        data.dropna(axis=0, inplace=True)
        return data

    

[文档]    def evaluate(self, tag_time: datetime, truth_file: dict, test_file: dict, **kwargs: Any) -> bool:
        """ 雷暴大风概率预报产品站点检验评估

        Args:
            tag_time : 当前时间
            truth_file : 真值数据信息。参数信息关键字参考CMAE config。
            test_file : 待测数据信息。参数信息关键字参考CMAE config。

        Keyword Args:
            file(str): 检验评估产品保存路径规则。
            timefcst(list): 预报时长表。涵盖预报产品所有的预报时长的列表。
            key(list): 默认为 ``Windstorm``。检验评估的变量关键字。
            threshold(list): 默认为 ``[0, 50, 101]``。雷暴大风概率预报阈值表。
            location(dict or None): 默认为 ``None``。检验评估经纬度范围。
            code(str): 默认为 ``WINDSTORM_03H_STATION``。检验评估对象的名称或代号。
            max_counter(int): 默认为 ``10``。 数据产品检验评估错误时，循环等待的次数。
            area(str): 默认为 ``None``。 检验区域名称，检验评估产品字段。

        Returns:
            是否正常完成评估。True 正常完成评估; False 未正常完成评估，或因路径错误。
        """
        # Keyword Args
        save = kwargs.pop("file")
        timefcst = kwargs.pop("timefcst")
        key = kwargs.get("key", "Windstorm")
        threshold = kwargs.get("threshold", [0, 50, 101])
        loc = kwargs.get("location", None)
        code = kwargs.get("code", "WINDSTORM_03H_STATION")
        max_counter = kwargs.get("max_counter", 10)
        area = kwargs.get("area", None)
        
        # 预处理
        if loc is not None and isinstance(loc, dict):
            for _ in ["slat", "slon", "elat", "elon"]:
                setattr(General, _.upper(), loc.pop(_))
        
        # 站点数据基础信息关键字
        stainfo = ["Station_Name", "Province", "City", "Cnty", "Station_Id_C",
                   truth_file["name_Lat"], truth_file["name_Lon"]]
        # 站点目标变量关键字
        stavars = ["WIN_S_Max", "WIN_S_Avg_2mi"]
        
        for lead_time in timefcst:
            fcst_time = tag_time + lead_time
            #  读取 truth 数据 >>> UTC
            path = Path(DataReader.catch(self.transform_time(fcst_time, timezone=truth_file["timezone"]),
                                         lead_time=None, file=truth_file["file"]))
            if path.exists():
                truth = pd.read_csv(path, sep="\t", low_memory=False)
                flag_lat = (General.SLAT - 0.1 <= truth.Lat) & (truth.Lat <= General.ELAT + 0.1)
                flag_lon = (General.SLON - 0.1 <= truth.Lon) & (truth.Lon <= General.ELON + 0.1)
                truth = truth[stainfo + stavars][flag_lat & flag_lon]
                max_flag = (17.2 <= truth.WIN_S_Max) & (truth.WIN_S_Max <= 100)  # 最大风速 ≥17.2 m/s
                avg_flag = (10.8 <= truth.WIN_S_Avg_2mi) & (truth.WIN_S_Avg_2mi <= 100)  # 平均风速 ≥10.8 m/s
                truth.loc[max_flag | avg_flag, "Windstorm"] = 90
                truth.loc[~max_flag & ~avg_flag, "Windstorm"] = 0
                
                # 读取闪电数据
                lpd = pd.DataFrame(columns=["经度", "纬度"])
                path = fcst_time.strftime(
                    "/mnt/PRESKY/project/Dianwang/other_data/gwth/data/LPD_LOCATE/%Y/%Y%m/%Y%m%d/Z_UPAR_C_BABJ_%Y%m%d%H*_P_LPD_LOCATE.txt")
                paths = glob.glob(path)
                if len(paths):
                    for path in paths:
                        tmp = self.read_lpd(path)
                        lpd = pd.concat[lpd, tmp]
                    lpd["经度"], lpd["纬度"] = (lpd["经度"].astype(float), lpd["纬度"].astype(float))
                    # 缩小闪电范围数据，便于后续数据提取
                    flag_lat = (General.SLAT - 0.2 <= lpd["纬度"]) & (lpd["纬度"] <= General.ELAT + 0.2)
                    flag_lon = (General.SLON - 0.2 <= lpd["经度"]) & (lpd["经度"] <= General.ELON + 0.2)
                    lpd = lpd[flag_lat & flag_lon]
                    for col in truth[truth.Windstorm == 100].itertuples():
                        flag_lat = (col.Lat - 0.2 <= lpd["纬度"]) & (lpd["纬度"] <= col.Lat + 0.2)
                        flag_lon = (col.Lon - 0.2 <= lpd["经度"]) & (lpd["经度"] <= col.Lon + 0.2)
                        # 半径范围没有闪电，则对应的大风不属于雷暴大风
                        if np.sum(flag_lat & flag_lon) == 0:
                            truth.loc[col.Index, key] = 0
                else:
                    truth.loc[:, key] = 0
            elif self.counter <= max_counter:
                Log.warning(self.logger, f"Hourly Data is missing, {path}!!!")
                return False
            else:
                Log.warning(self.logger, f"Hourly Data is missing, {path}!!!   !!!And go on evaluate...!!!")
                truth = pd.DataFrame(columns=stainfo + stavars)
            truth.reset_index(drop=True, inplace=True)
            
            #  读取 test 数据 >>> UTC
            path = Path(DataReader.catch(time=self.transform_time(tag_time, timezone=test_file["timezone"]),
                                         lead_time=lead_time, file=test_file["file"]))
            if path.exists():
                Log.info(self.logger, f"Success catch the WindStorm Product: {path}")
            elif self.counter <= max_counter:
                Log.warning(self.logger, f"WindStorm Product is missing, {path}!!!")
                return False
            else:
                Log.warning(self.logger, f"WindStorm Product is missing, {path}!!!   !!!And go on evaluate...!!!")
            test = DataReader.load(path, vnames=[key], name_Lat=test_file["name_Lat"], name_Lon=test_file["name_Lon"])
            
            # Evaluate
            res = dict(stime=tag_time.strftime("%Y%m%d%H%M"), ftime=fcst_time.strftime("%Y%m%d%H%M"),
                       area=area, type=code, method="TS",
                       PO=Default.MissValue, FAR=Default.MissValue,
                       TS=Default.MissValue, POD=Default.MissValue)
            if (key in test) and len(truth):
                new_test = np.ones(len(truth)) * Default.MissValue
                for col in truth.itertuples():
                    nx = np.where((col.Lon - 0.1 <= test["Lon"]) & (test["Lon"] <= col.Lon + 0.1))[0]
                    ny = np.where((col.Lat - 0.1 <= test["Lat"]) & (test["Lat"] <= col.Lat + 0.1))[0]
                    nx, ny = np.meshgrid(nx, ny)
                    new_test[col.Index] = np.max(test["Windstorm"][ny, nx])
                
                tmp = Score.scores(truth=truth[key], test=new_test, threshold=threshold,
                                   names=["TS", "PO", "POD", "FAR"])
                for name in ["TS", "PO", "POD", "FAR"]:
                    res[name] = tmp["1"][name]
            
            # 保存数据
            save_path = Path(DataReader.merge_file(time=tag_time, lead_time=lead_time, file=save))
            save_path.parent.mkdir(parents=True, exist_ok=True)
            with open(save_path, "w", encoding="utf8") as fp:
                fp.write(json.dumps(res, ensure_ascii=False) + "\n")
            
            Log.info(self.logger, f">>>TS Check Result: {save_path}")
        return True
