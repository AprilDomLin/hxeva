from .base import BaseProdEvaluate, DataReader, Log
from ..config import Default, General
from ..evaluate import Score

__all__ = ["HailFcstSta"]



[文档]class HailFcstSta(BaseProdEvaluate, ABC):
    """ 冰雹概率预报产品站点检验评估
    
    Args:
        name : 默认为 ``Hail``。产品名称。
        log_root : 默认为 ``./logs`` 。日志文件夹根目录
        extend : 默认为 ``None`` 。经纬度范围。
    """
    
    def __init__(self, name="Hail", log_root="./logs", extend=None):
        super(HailFcstSta, self).__init__(name=name, log_root=log_root)
        if isinstance(extend, dict):
            General.SLAT = extend["slat"]
            General.SLON = extend["slon"]
            General.ELAT = extend["elat"]
            General.ELON = extend["elon"]
    

[文档]    def evaluate(self, tag_time: datetime, truth_file: dict, test_file: dict, **kwargs: Any) -> bool:
        """ 冰雹概率预报产品站点检验评估
        
        Args:
            tag_time : 当前时间
            truth_file : 真值数据信息。参数信息关键字参考CMAE config。
            test_file : 待测数据信息。参数信息关键字参考CMAE config。
        
        Keyword Args:
            file(str): 检验评估产品保存路径规则。
            timefcst(list): 预报时长表。涵盖预报产品所有的预报时长的列表。
            key(list): 默认为 ``Hail``。检验评估的变量关键字。
            threshold(list): 默认为 ``[0, 50, 101]``。冰雹概率预报阈值表。
            location(dict or None): 默认为 ``None``。检验评估经纬度范围。
            code(str): 默认为 ``HAIL_03H_STATION``。检验评估对象的名称或代号。
            max_counter(int): 默认为 ``10``。 数据产品检验评估错误时，循环等待的次数。
            area(str): 默认为 ``None``。 检验区域名称，检验评估产品字段。

        Returns:
            是否正常完成评估。True 正常完成评估; False 未正常完成评估，或因路径错误。
        """
        # Keyword Args
        save = kwargs.pop("file")
        timefcst = kwargs.pop("timefcst")
        key = kwargs.get("key", "Hail")
        threshold = kwargs.get("threshold", [0, 50, 101])
        loc = kwargs.get("location", None)
        code = kwargs.get("code", "HAIL_03H_STATION")
        max_counter = kwargs.get("max_counter", 10)
        area = kwargs.get("area", None)
        
        # 预处理
        if loc is not None and isinstance(loc, dict):
            for _ in ["slat", "slon", "elat", "elon"]:
                setattr(General, _.upper(), loc.pop(_))
        
        # 站点数据基础信息关键字
        stainfo = ["Station_Name", "Province", "City", "Cnty", "Datetime", "Station_Id_C",
                   truth_file["name_Lat"], truth_file["name_Lon"]]
        # 站点目标变量关键字
        stavars = ["HAIL_OTime", "Hail"]
        
        # 读取冰雹站点数据
        path = Path(DataReader.catch(self.transform_time(tag_time, timezone=truth_file["timezone"]),
                                     lead_time=None, file=truth_file["file"]))
        if path.exists():
            # 日志站点数据存在时，读取冰雹数据
            Log.info(self.logger, f"Catch the daily path: {path}")
            obs_tmp = pd.read_csv(path, sep="\t", low_memory=False)
            obs_tmp = obs_tmp[stainfo + stavars][obs_tmp.Hail == 1]
            flagLat = (General.SLAT - 0.1 <= obs_tmp.Lat) & (obs_tmp.Lat <= General.ELAT + 0.1)
            flagLon = (General.SLON - 0.1 <= obs_tmp.Lon) & (obs_tmp.Lon <= General.ELON + 0.1)
            obs_tmp = obs_tmp[flagLat & flagLon]
            obs_tmp = obs_tmp[~obs_tmp.HAIL_OTime.isin([999999, "999999", "999998"])]
        elif self.counter <= max_counter:
            Log.warning(self.logger, f"Daily Data is missing, {path}!!!")
            return False
        else:
            Log.warning(self.logger, f"Daily Data is missing, {path}!!!   !!!And go on evaluate...!!!")
            obs_tmp = pd.DataFrame(columns=stainfo + stavars + ["stime", "etime"])
        
        obs = pd.DataFrame(columns=stainfo + ["stime", "etime", key])
        index = 0
        if len(obs_tmp):
            for col in obs_tmp.itertuples():
                dateTime = datetime.strptime(getattr(col, "Datetime"), "%Y-%m-%d %H:%M:%S")
                dTimes = str(getattr(col, "HAIL_OTime")).split("''")
                for dTime in dTimes:
                    # Basic Information
                    for _ in stainfo:
                        obs.loc[index, _] = getattr(col, _)
                    #  冰雹观测产品转化为发生概率值
                    obs.loc[index, key] = 90
                    # Hail Time
                    if len(dTime) <= 4:
                        startTime = dateTime + timedelta(minutes=int(dTime[:-2]) * 60 + int(dTime[-2:]))
                        endTime = startTime
                    elif len(dTime) == 8:
                        startTime = dateTime + timedelta(minutes=int(dTime[:2]) * 60 + int(dTime[2:4]))
                        endTime = dateTime + timedelta(minutes=int(dTime[4:6]) * 60 + int(dTime[6:8]))
                    else:
                        startTime = dateTime + timedelta(minutes=int(dTime[:1]) * 60 + int(dTime[1:3]))
                        endTime = dateTime + timedelta(minutes=int(dTime[3:-2]) * 60 + int(dTime[-2:]))
                    # 客户需求，时间窗口前移30min（算法识别），后移18min（三个体扫时间）
                    obs.loc[index, "stime"] = startTime - timedelta(minutes=30)
                    obs.loc[index, "etime"] = endTime + timedelta(minutes=18)
                    index += 1
        
        for hour in range(24):
            rep_time = tag_time + timedelta(hours=hour)
            for lead_time in timefcst:
                fcst_time = rep_time + lead_time
                #  筛选 truth 数据 >>> UTC
                truth = obs[(rep_time <= obs.etime) & (obs.stime <= fcst_time)]
                truth.reset_index(drop=True, inplace=True)
                
                #  读取 test 数据 >>> UTC
                path = Path(DataReader.catch(time=self.transform_time(tag_time, timezone=test_file["timezone"]),
                                             lead_time=lead_time, file=test_file["file"]))
                if path.exists():
                    Log.info(self.logger, f"Success catch the Hail Product: {path}")
                elif self.counter <= max_counter:
                    Log.warning(self.logger, f"Hail Product is missing, {path}!!!")
                    return False
                else:
                    Log.warning(self.logger, f"Hail Product is missing, {path}!!!   !!!And go on evaluate...!!!")
                test = DataReader.load(path, vnames=[key], name_Lat=test_file["name_Lat"],
                                       name_Lon=test_file["name_Lon"])
                
                # Evaluate
                # 初始化检验结果
                res = dict(stime=rep_time.strftime("%Y%m%d%H%M"), ftime=fcst_time.strftime("%Y%m%d%H%M"),
                           area=area, type=code, method="TS",
                           PO=Default.MissValue, FAR=Default.MissValue,
                           TS=Default.MissValue, POD=Default.MissValue)
                if (key in test) and len(truth):
                    new_test = np.ones(len(truth)) * Default.MissValue
                    for col in truth.itertuples():
                        nx = np.where((col.Lon - 0.1 <= test["Lon"]) & (test["Lon"] <= col.Lon + 0.1))[0]
                        ny = np.where((col.Lat - 0.1 <= test["Lat"]) & (test["Lat"] <= col.Lat + 0.1))[0]
                        new_test[col.Index] = np.max(test[key][ny, nx])
                    
                    tmp = Score.scores(truth=truth[key], test=new_test, threshold=threshold,
                                       names=["TS", "PO", "POD", "FAR"])
                    for name in ["TS", "PO", "POD", "FAR"]:
                        res[name] = tmp["1"][name]
                
                # 保存数据
                save_path = Path(DataReader.merge_file(time=rep_time, lead_time=lead_time, file=save))
                save_path.parent.mkdir(parents=True, exist_ok=True)
                with open(save_path, "w", encoding="utf8") as fp:
                    fp.write(json.dumps(res, ensure_ascii=False) + "\n")
                
                Log.info(self.logger, f">>>TS Check Result: {save_path}")
        return True
